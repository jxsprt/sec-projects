SSL/TLS Certificate Inspector

Fetch and analyze SSL/TLS certificates — check expiry, issuer chain, cipher suites, and security configuration.

python ssl tls certificates

Learning Objectives

  • Understand the X.509 certificate structure and how TLS trust chains work
  • Use Python's ssl and cryptography modules to retrieve and parse certificates
  • Identify certificate problems: expiry, weak signature algorithms (SHA-1), small key sizes
  • Learn to spot self-signed, wildcard, and multi-domain (SAN) certificates
  • Build a practical tool for security audits and reconnaissance

Prerequisites

  • Python 3.8+ (standard library ssl module is built-in)
  • The cryptography package for extended parsing (pip install cryptography)
  • Basic understanding of what SSL/TLS certificates are and why HTTPS matters

Step 1: Fetch a Server's Certificate

Connect to any TLS-enabled server and retrieve its certificate chain. Python's ssl module makes this straightforward.

import ssl
import socket
from datetime import datetime, timezone

def get_certificate(hostname: str, port: int = 443, timeout: float = 5.0) -> dict:
    """Connect to a server and retrieve its SSL/TLS certificate."""

    context = ssl.create_default_context()
    # Don't verify — we want to inspect even bad certs
    context.check_hostname = False
    context.verify_mode = ssl.CERT_NONE

    with socket.create_connection((hostname, port), timeout=timeout) as sock:
        with context.wrap_socket(sock, server_hostname=hostname) as ssock:
            cert_bin = ssock.getpeercert(binary_form=True)
            cert_dict = ssock.getpeercert()  # Parsed form
            cipher = ssock.cipher()           # (name, version, bits)

    return {
        "hostname": hostname,
        "port": port,
        "cert_der": cert_bin,
        "cert_dict": cert_dict,
        "cipher": cipher,
    }

# Test it
result = get_certificate("example.com")
print(f"Connected to {result['hostname']}:{result['port']}")
print(f"Cipher: {result['cipher'][0]} ({result['cipher'][2]} bits)")
print(f"Subject: {dict(x[0] for x in result['cert_dict']['subject'])}")
print(f"Issuer:  {dict(x[0] for x in result['cert_dict']['issuer'])}")

Step 2: Parse and Analyze Certificate Details

Use the cryptography library to decode the DER binary and extract rich certificate information — validity dates, SANs, key usage, and more.

from cryptography import x509
from cryptography.x509.oid import NameOID, ExtensionOID
from cryptography.hazmat.primitives import hashes

def analyze_certificate(cert_der: bytes) -> dict:
    """Parse a DER-encoded certificate and extract key details."""

    cert = x509.load_der_x509_certificate(cert_der)

    # Subject
    subject_cn = cert.subject.get_attributes_for_oid(NameOID.COMMON_NAME)
    cn = subject_cn[0].value if subject_cn else "N/A"

    # Issuer
    issuer_cn = cert.issuer.get_attributes_for_oid(NameOID.COMMON_NAME)
    issuer = issuer_cn[0].value if issuer_cn else "N/A"

    # Validity
    now = datetime.now(timezone.utc)
    days_left = (cert.not_valid_after_utc - now).days

    # Subject Alternative Names (SANs)
    sans = []
    try:
        san_ext = cert.extensions.get_extension_for_oid(
            ExtensionOID.SUBJECT_ALTERNATIVE_NAME
        )
        sans = san_ext.value.get_values_for_type(x509.DNSName)
    except x509.ExtensionNotFound:
        pass

    # Signature algorithm
    sig_alg = cert.signature_algorithm_oid._name

    # Fingerprints
    fingerprint_sha1 = cert.fingerprint(hashes.SHA1()).hex()
    fingerprint_sha256 = cert.fingerprint(hashes.SHA256()).hex()

    # Key size
    public_key = cert.public_key()
    key_size = public_key.key_size

    # Serial number
    serial = hex(cert.serial_number)

    return {
        "cn": cn,
        "issuer": issuer,
        "serial": serial,
        "not_before": cert.not_valid_before_utc,
        "not_after": cert.not_valid_after_utc,
        "days_left": days_left,
        "self_signed": (cert.issuer == cert.subject),
        "sans": sans,
        "signature_algorithm": sig_alg,
        "key_size": key_size,
        "sha1_fingerprint": fingerprint_sha1,
        "sha256_fingerprint": fingerprint_sha256,
    }

# Test
result = get_certificate("google.com")
info = analyze_certificate(result["cert_der"])
print(f"CN: {info['cn']}")
print(f"Issuer: {info['issuer']}")
print(f"Valid: {info['not_before'].strftime('%Y-%m-%d')} to {info['not_after'].strftime('%Y-%m-%d')}")
print(f"Days left: {info['days_left']}")
print(f"Key: {info['key_size']}-bit, Sig: {info['signature_algorithm']}")
print(f"SANs: {', '.join(info['sans'][:5])}...")
print(f"Self-signed: {info['self_signed']}")

Step 3: Security Grade Assessment

Build logic to flag security issues — expired certificates, weak keys, deprecated algorithms, and missing features.

from enum import Enum

class Severity(Enum):
    OK = "✅"
    WARN = "⚠️"
    FAIL = "❌"

def assess_security(info: dict) -> list:
    """Analyze certificate and return a list of security findings."""

    findings = []

    # Expiry check
    if info["days_left"] < 0:
        findings.append((Severity.FAIL, f"Certificate EXPIRED {abs(info['days_left'])} days ago"))
    elif info["days_left"] < 30:
        findings.append((Severity.WARN, f"Certificate expires in {info['days_left']} days"))
    elif info["days_left"] < 90:
        findings.append((Severity.OK, f"Certificate expires in {info['days_left']} days — renew soon"))
    else:
        findings.append((Severity.OK, f"Certificate valid for {info['days_left']} days"))

    # Key size check (NIST recommends 2048+ for RSA)
    if info["key_size"] < 2048:
        findings.append((Severity.FAIL, f"Key size {info['key_size']} bits is INSECURE (min 2048)"))
    elif info["key_size"] < 4096:
        findings.append((Severity.OK, f"Key size {info['key_size']} bits meets minimum"))
    else:
        findings.append((Severity.OK, f"Key size {info['key_size']} bits — strong"))

    # Signature algorithm
    if "sha1" in info["signature_algorithm"].lower():
        findings.append((Severity.FAIL, f"Weak signature: {info['signature_algorithm']} (SHA-1 is broken)"))
    elif "sha256" in info["signature_algorithm"].lower() or "sha384" in info["signature_algorithm"].lower():
        findings.append((Severity.OK, f"Signature algorithm: {info['signature_algorithm']}"))
    elif "md5" in info["signature_algorithm"].lower():
        findings.append((Severity.FAIL, f"MD5 signature: {info['signature_algorithm']} — completely broken!"))
    else:
        findings.append((Severity.WARN, f"Signature algorithm: {info['signature_algorithm']}"))

    # Self-signed
    if info["self_signed"]:
        findings.append((Severity.WARN, "Self-signed certificate — not trusted by browsers"))

    # SAN check (modern certs should have SANs)
    if not info["sans"]:
        findings.append((Severity.WARN, "No Subject Alternative Names — browsers may reject"))

    return findings

# Test against Let's Encrypt + a known bad cert
for host in ["letsencrypt.org", "expired.badssl.com"]:
    try:
        result = get_certificate(host)
        info = analyze_certificate(result["cert_der"])
        print(f"\n{'='*60}")
        print(f"  {host} — {info['cn']}")
        print(f"{'='*60}")
        for severity, msg in assess_security(info):
            print(f"  {severity.value} {msg}")
    except Exception as e:
        print(f"  ❌ Error for {host}: {e}")

Step 4: Build the Complete Inspector CLI

Wire everything together with argparse for a professional CLI tool. Support single-target analysis and bulk scanning from a file.

import argparse
import sys

def inspect_host(hostname: str, port: int = 443) -> None:
    """Fetch, parse, analyze, and pretty-print certificate info."""

    try:
        result = get_certificate(hostname, port)
        info = analyze_certificate(result["cert_der"])
    except socket.timeout:
        print(f"❌ {hostname}:{port} — Connection timed out")
        return
    except ssl.SSLError as e:
        print(f"❌ {hostname}:{port} — SSL error: {e}")
        return
    except Exception as e:
        print(f"❌ {hostname}:{port} — Error: {e}")
        return

    print(f"\n{'='*60}")
    print(f"  {hostname}:{port}")
    print(f"  Subject: {info['cn']}")
    print(f"  Issuer:  {info['issuer']}")
    print(f"  Serial:  {info['serial']}")
    print(f"  Valid:   {info['not_before'].strftime('%Y-%m-%d')} → {info['not_after'].strftime('%Y-%m-%d')}")
    print(f"  Cipher:  {result['cipher'][0]} ({result['cipher'][2]} bits)")
    print(f"  Key:     {info['key_size']}-bit | Sig: {info['signature_algorithm']}")
    print(f"  SHA256:  {info['sha256_fingerprint']}")
    if info['sans']:
        print(f"  SANs:    {', '.join(info['sans'][:8])}")
    print(f"{'='*60}")

    for severity, msg in assess_security(info):
        print(f"  {severity.value} {msg}")

def main():
    parser = argparse.ArgumentParser(
        description="SSL/TLS Certificate Inspector — analyze certs for security issues"
    )
    parser.add_argument("hosts", nargs="+", help="Hostname(s) to inspect")
    parser.add_argument("-p", "--port", type=int, default=443,
                        help="Port (default: 443)")
    parser.add_argument("-f", "--file", help="Read hostnames from file (one per line)")

    args = parser.parse_args()

    hosts = args.hosts
    if args.file:
        with open(args.file) as f:
            hosts.extend(line.strip() for line in f if line.strip())

    for host in hosts:
        inspect_host(host, args.port)

if __name__ == "__main__":
    main()

Testing Your Inspector

Verify against known-good and known-bad test sites:

  • badssl.com: A fantastic QA resource — test expired.badssl.com, self-signed.badssl.com, sha1-intermediate.badssl.com, 1000-sans.badssl.com
  • google.com / cloudflare.com: Should return clean reports with modern configs
  • Compare with SSL Labs: Run SSL Labs on the same host and compare findings
  • localhost scan: If you have a local web server, test against it

Extension Ideas

  • TLS version checking: Test which TLS versions a server supports (1.0, 1.1 should fail; 1.2, 1.3 should pass)
  • Certificate chain validation: Build the full chain to a trusted root and verify every signature in the chain
  • CT log search: Query crt.sh for Certificate Transparency logs — discover all certs ever issued for a domain
  • JSON/HTML report: Generate machine-readable JSON or a styled HTML report for compliance audits
  • Monitoring mode: Check certificates daily via cron and alert before they expire
