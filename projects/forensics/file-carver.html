File Carver

Recover deleted or hidden files from disk images by searching for file headers/magic bytes and reconstructing files.

python binary-analysis file-headers forensics

Learning Objectives

  • Understand how file systems reference data and why "deleted" files remain recoverable
  • Identify file types by their magic bytes (header signatures) at known byte offsets
  • Implement header/footer carving — find the start of a file and scan forward until the end marker
  • Handle common image, document, and archive formats (JPEG, PNG, PDF, ZIP, GIF)
  • Validate carved files using structural checks to reduce false positives

Prerequisites

  • Python 3.8+ (standard library only)
  • Understanding of hexadecimal and binary file formats
  • A small raw disk image or binary blob for testing

Step 1: Magic Bytes Database & Header Scanner

Build the magic bytes database and a scanner that finds file headers in raw binary data. Each file format has a unique signature at the beginning of the file.

#!/usr/bin/env python3
"""File Carver — recover files from raw disk images using header/footer signatures."""

import os
import argparse
from collections import namedtuple

# File signatures: (header_bytes, footer_bytes, extension, description)
# Headers are typically at offset 0 of the file
FileSignature = namedtuple("FileSignature", ["header", "footer", "ext", "description"])

SIGNATURES = [
    # Images
    FileSignature(b"\xff\xd8\xff",          b"\xff\xd9",       "jpg",  "JPEG Image"),
    FileSignature(b"\x89PNG\r\n\x1a\n",      b"IEND\xae\x42\x60\x82", "png", "PNG Image"),
    FileSignature(b"GIF89a",                b"\x00\x3b",       "gif",  "GIF Image (89a)"),
    FileSignature(b"GIF87a",                b"\x00\x3b",       "gif",  "GIF Image (87a)"),
    FileSignature(b"BM",                    None,              "bmp",  "BMP Image"),
    FileSignature(b"\x00\x00\x01\x00",      None,              "ico",  "ICO Icon"),

    # Documents
    FileSignature(b"%PDF",                  b"%%EOF",          "pdf",  "PDF Document"),
    FileSignature(b"\xd0\xcf\x11\xe0",      None,              "doc",  "MS Office OLE2 (DOC/XLS/PPT)"),
    FileSignature(b"PK\x03\x04",            None,              "zip",  "ZIP Archive / DOCX / XLSX"),

    # Archives
    FileSignature(b"\x1f\x8b\x08",          None,              "gz",   "GZip Archive"),
    FileSignature(b"Rar!\x1a\x07\x00",      None,              "rar",  "RAR Archive"),
    FileSignature(b"\x37\x7a\xbc\xaf\x27\x1c", None,           "7z",   "7-Zip Archive"),

    # Executables
    FileSignature(b"MZ",                    None,              "exe",  "Windows PE / DOS Executable"),
    FileSignature(b"\x7fELF",               None,              "elf",  "Linux ELF Binary"),

    # Multimedia
    FileSignature(b"\x00\x00\x01\xba",      None,              "mpg",  "MPEG Video"),
    FileSignature(b"ID3",                   None,              "mp3",  "MP3 Audio"),
    FileSignature(b"RIFF",                  None,              "wav",  "WAV Audio / AVI Video"),
]

class HeaderScanner:
    """Scan raw bytes for file header signatures."""

    def __init__(self, signatures=None):
        self.signatures = signatures or SIGNATURES

    def scan(self, data):
        """Find all header offsets in binary data. Returns list of (offset, signature)."""
        hits = []
        for sig in self.signatures:
            offset = 0
            while True:
                pos = data.find(sig.header, offset)
                if pos == -1:
                    break
                hits.append((pos, sig))
                offset = pos + 1  # continue search after this hit
        return sorted(hits, key=lambda h: h[0])

    def scan_file(self, filepath):
        """Memory-map a file and scan for headers (handles large files)."""
        file_size = os.path.getsize(filepath)
        print(f"[*] Scanning {filepath} ({file_size:,} bytes)...")

        hits = []
        # Use mmap for large files to avoid loading everything into RAM
        try:
            import mmap
            with open(filepath, "rb") as f:
                with mmap.mmap(f.fileno(), 0, access=mmap.ACCESS_READ) as mm:
                    data = mm
                    hits = self.scan(data)
        except (OSError, ValueError):
            # Fallback: read in chunks
            with open(filepath, "rb") as f:
                data = f.read()
            hits = self.scan(data)

        print(f"[+] Found {len(hits)} header(s)")
        for offset, sig in hits:
            print(f"  [{sig.ext:>5}] offset={offset:>12,}  {sig.description}")
        return hits

Step 2: File Carver with Footer Detection

For each header found, scan forward to find the corresponding footer (if one exists) and carve the bytes between them. If no footer is defined, carve a fixed maximum length or until the next header.

class FileCarver:
    """Carve files from raw data using header/footer matching."""

    def __init__(self, max_carve_size=50 * 1024 * 1024):  # 50 MB default max
        self.max_carve_size = max_carve_size
        self.scanner = HeaderScanner()

    def carve(self, data, base_offset=0):
        """Carve files from data. Returns list of (filename, bytes, signature)."""
        hits = self.scanner.scan(data)
        if not hits:
            return []

        recovered = []

        for i, (offset, sig) in enumerate(hits):
            start = offset
            end = None

            # Determine end of file
            if sig.footer:
                # Search for footer after header
                footer_pos = data.find(sig.footer, start + len(sig.header))
                if footer_pos != -1:
                    end = footer_pos + len(sig.footer)
                else:
                    # No footer found — use next header or max carve size
                    if i + 1 < len(hits):
                        end = min(hits[i + 1][0], start + self.max_carve_size)
                    else:
                        end = min(len(data), start + self.max_carve_size)
            else:
                # No footer defined — carve until next header or max size
                if i + 1 < len(hits):
                    next_start = hits[i + 1][0]
                    end = min(next_start, start + self.max_carve_size)
                else:
                    end = min(len(data), start + self.max_carve_size)

            carved_bytes = data[start:end]
            recovered.append((sig, carved_bytes, start))

        return recovered

    def carve_file(self, source_path, output_dir="carved"):
        """Carve files from a disk image file on disk."""
        os.makedirs(output_dir, exist_ok=True)

        file_size = os.path.getsize(source_path)
        print(f"[*] Carving from {source_path} ({file_size:,} bytes)")

        with open(source_path, "rb") as f:
            data = f.read()

        results = self.carve(data)
        saved = 0

        for i, (sig, file_bytes, offset) in enumerate(results):
            filename = os.path.join(output_dir, f"carved_{offset:08d}.{sig.ext}")
            with open(filename, "wb") as out:
                out.write(file_bytes)

            size_kb = len(file_bytes) / 1024
            print(f"  [+] {filename}  ({size_kb:.1f} KB)  [{sig.description}]")
            saved += 1

        print(f"\n[*] Carved {saved} file(s) to {output_dir}/")
        return saved

    def carve_with_validation(self, source_path, output_dir="carved"):
        """Carve and validate — check carved files have sane internal structure."""
        os.makedirs(output_dir, exist_ok=True)

        with open(source_path, "rb") as f:
            data = f.read()

        results = self.carve(data)
        valid = 0
        corrupt = 0

        for i, (sig, file_bytes, offset) in enumerate(results):
            if self._validate(sig, file_bytes):
                status = "VALID"
                filename = os.path.join(output_dir, f"carved_{offset:08d}.{sig.ext}")
                valid += 1
            else:
                status = "CORRUPT"
                filename = os.path.join(output_dir, f"corrupt_{offset:08d}.{sig.ext}")
                corrupt += 1

            with open(filename, "wb") as out:
                out.write(file_bytes)

            size_kb = len(file_bytes) / 1024
            print(f"  [{status:>7}] {os.path.basename(filename)}  ({size_kb:.1f} KB)  [{sig.description}]")

        print(f"\n[*] Done: {valid} valid, {corrupt} corrupt out of {valid + corrupt} total")
        return valid + corrupt

    def _validate(self, sig, data):
        """Quick structural validation checks per file type."""
        if sig.ext == "png":
            # Check for IEND chunk at the end
            return data.endswith(sig.footer) if sig.footer else len(data) > 100
        if sig.ext == "jpg":
            # JPEG should end with FF D9
            return data.endswith(b"\xff\xd9")
        if sig.ext == "pdf":
            return data.rstrip().endswith(b"%%EOF")
        if sig.ext == "gif":
            return data.endswith(b"\x00\x3b") or data.endswith(b";")
        if sig.ext == "zip":
            # ZIP has central directory at the end
            return b"PK\x05\x06" in data[-100:] or b"PK\x01\x02" in data[-100:]
        return True  # can't easily validate other types

Step 3: Complete CLI Carving Tool

Combine scanner, carver, and validation into a single tool with progress and reporting.

#!/usr/bin/env python3

import argparse
import json
import os
import time

def main():
    parser = argparse.ArgumentParser(
        description="File Carver — Recover deleted files from disk images"
    )
    parser.add_argument("source", help="Disk image or raw binary file to carve")
    parser.add_argument("-o", "--output", default="carved", help="Output directory (default: carved/)")
    parser.add_argument("--max-size", type=int, default=50,
                        help="Maximum carve size in MB (default: 50)")
    parser.add_argument("--validate", action="store_true",
                        help="Validate carved files and separate corrupt ones")
    parser.add_argument("--types", help="Only carve specific types (comma-separated: jpg,png,pdf)")
    parser.add_argument("--report", help="Save carving report as JSON")
    parser.add_argument("--list", action="store_true",
                        help="List supported file types and exit")
    args = parser.parse_args()

    if args.list:
        print("Supported file types:")
        for sig in SIGNATURES:
            print(f"  .{sig.ext:<5} {sig.description}")
        return

    if not os.path.exists(args.source):
        print(f"[-] File not found: {args.source}")
        return

    start_time = time.time()
    max_bytes = args.max_size * 1024 * 1024

    carver = FileCarver(max_carve_size=max_bytes)

    # Filter signatures if requested
    if args.types:
        types_set = set(args.types.split(","))
        carver.scanner.signatures = [s for s in SIGNATURES if s.ext in types_set]
        print(f"[*] Filtered to types: {', '.join(types_set)} ({len(carver.scanner.signatures)} signatures)")

    if args.validate:
        count = carver.carve_with_validation(args.source, args.output)
    else:
        count = carver.carve_file(args.source, args.output)

    elapsed = time.time() - start_time
    print(f"\n[*] Completed in {elapsed:.1f}s")

    if args.report:
        report = {"source": args.source, "files_carved": count, "elapsed_sec": elapsed}
        with open(args.report, "w") as f:
            json.dump(report, f, indent=2)
        print(f"[+] Report saved to {args.report}")


if __name__ == "__main__":
    main()

  • Use a forensic test image: Download the CFReDS test images from NIST
  • Verify with Foremost: Run foremost -i test.bin -o foremost_out and compare results
  • Test partial files: Truncate a JPEG at random points to test footer recovery
