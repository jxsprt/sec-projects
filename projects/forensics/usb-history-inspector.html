USB Device History Inspector

Parse Windows registry hives and Linux logs to enumerate connected USB devices — serials, timestamps, vendor IDs.

python registry forensics windows

Learning Objectives

  • Locate and parse Windows registry keys that record USB device connection history
  • Extract USB device serial numbers, vendor IDs (VID), product IDs (PID), and last connection timestamps
  • Cross-reference VID/PID against the USB ID database to identify manufacturer and device type
  • Parse Linux system logs (/var/log/syslog, dmesg) for USB insertion events
  • Build a timeline of USB device connections — useful for data exfiltration investigations

Prerequisites

  • Python 3.8+ (standard library — no external deps for basic operation)
  • Understanding of Windows registry structure (HKLM, HKCU, hives)
  • Access to exported registry hives or a live Windows system for testing
  • Optionally: pip install python-registry for direct hive parsing

Step 1: Registry Key Parser (from exported .reg files)

Step 2: Linux USB Log Parser

Parse Linux kernel logs and system logs for USB device insertion events. The kernel logs contain detailed VID/PID, serial numbers, and manufacturer strings.

import re
import subprocess
from datetime import datetime

class LinuxUSBLogParser:
    """Parse Linux system logs for USB connection events."""

    # USB ID database URL for reference
    USB_IDS_URL = "http://www.linux-usb.org/usb.ids"

    # Known vendor names (small subset — full list at linux-usb.org)
    VENDOR_NAMES = {
        "0781": "SanDisk", "0951": "Kingston", "13fe": "Kingston",
        "058f": "Alcor Micro", "090c": "Silicon Motion",
        "0bda": "Realtek", "048d": "Integrated Technology Express",
        "05e3": "Genesys Logic", "1f75": "Innostor",
        "1005": "Apacer", "1307": "Transcend",
        "8564": "Transcend", "03f0": "HP",
        "413c": "Dell", "046d": "Logitech",
        "8087": "Intel", "1050": "Yubico",
    }

    def __init__(self):
        self.events = []

    def parse_dmesg(self):
        """Parse kernel ring buffer for USB events."""
        try:
            output = subprocess.check_output(
                ["dmesg", "-T"], text=True, stderr=subprocess.DEVNULL
            )
        except (subprocess.CalledProcessError, FileNotFoundError):
            print("[-] Could not read dmesg (try with sudo)")
            return []

        return self._parse_log_lines(output.splitlines())

    def parse_log_file(self, log_path):
        """Parse a syslog file for USB events."""
        if not os.path.exists(log_path):
            print(f"[-] Log file not found: {log_path}")
            return []

        with open(log_path, "r", encoding="utf-8", errors="ignore") as f:
            lines = f.readlines()

        return self._parse_log_lines(lines)

    def _parse_log_lines(self, lines):
        """Extract USB device events from log lines."""
        # Patterns for USB mass storage devices
        usb_patterns = [
            # New USB device found: idVendor=XXXX, idProduct=XXXX
            re.compile(
                r"New USB device found.*?idVendor=([0-9a-f]{4}),\s*idProduct=([0-9a-f]{4})",
                re.IGNORECASE
            ),
            # Product/Manufacturer strings
            re.compile(
                r"Product:\s*(.+)$", re.IGNORECASE
            ),
            re.compile(
                r"Manufacturer:\s*(.+)$", re.IGNORECASE
            ),
            # Serial number
            re.compile(
                r"SerialNumber:\s*(\S+)", re.IGNORECASE
            ),
            # USB Mass Storage device detected
            re.compile(
                r"usb-storage.*?detected", re.IGNORECASE
            ),
            # SCSI removable disk (sdX becomes /dev/sdX)
            re.compile(
                r"sd\s(\d+:\d+:\d+:\d+).*?\[(sd[a-z]+)\].*?Attached SCSI removable disk",
                re.IGNORECASE
            ),
        ]

        current_device = {}
        timestamp = None

        for line in lines:
            # Try to parse timestamp from syslog format: "Jun 18 10:30:45 hostname kernel:"
            ts_match = re.match(
                r"(\w{3}\s+\d{1,2}\s+\d{2}:\d{2}:\d{2})",
                line
            )
            if ts_match:
                try:
                    ts_str = ts_match.group(1)
                    # Assume current year since syslog doesn't include it
                    timestamp = datetime.strptime(
                        f"{datetime.now().year} {ts_str}",
                        "%Y %b %d %H:%M:%S"
                    )
                except ValueError:
                    pass

            # Extract VID/PID
            vidpid_match = usb_patterns[0].search(line)
            if vidpid_match:
                vid = vidpid_match.group(1)
                pid = vidpid_match.group(2)
                current_device = {
                    "vid": vid,
                    "pid": pid,
                    "vendor_name": self.VENDOR_NAMES.get(vid, f"Unknown ({vid})"),
                    "time": timestamp,
                    "raw_line": line.strip(),
                }
                self.events.append(current_device)
                continue

            # Extract Product name
            if current_device and "product" not in current_device:
                prod_match = usb_patterns[1].search(line)
                if prod_match:
                    current_device["product"] = prod_match.group(1).strip()

            # Extract Serial
            if current_device and "serial" not in current_device:
                serial_match = usb_patterns[3].search(line)
                if serial_match:
                    current_device["serial"] = serial_match.group(1)

        return self.events

    def read_usb_ids(self, usb_ids_path="/usr/share/misc/usb.ids"):
        """Parse the USB ID database file for vendor/product names."""
        vendor_map = {}
        if not os.path.exists(usb_ids_path):
            return vendor_map

        current_vendor = None
        with open(usb_ids_path, "r", encoding="utf-8", errors="ignore") as f:
            for line in f:
                line = line.rstrip()
                if not line or line.startswith("#"):
                    continue

                if not line.startswith("\t"):  # Vendor line
                    parts = line.split(" ", 1)
                    if len(parts) == 2:
                        vendor_map[parts[0].lower()] = parts[1]
                        current_vendor = parts[0].lower()
                elif current_vendor and line.startswith("\t") and not line.startswith("\t\t"):
                    # Product line
                    parts = line.strip().split(" ", 1)
                    if len(parts) == 2:
                        key = f"{current_vendor}:{parts[0]}"
                        vendor_map[key] = parts[1]

        return vendor_map

    def report(self):
        """Print USB events in chronological order."""
        print(f"\n{'='*60}")
        print(f"  LINUX USB DEVICE HISTORY")
        print(f"{'='*60}")
        print(f"  Total events: {len(self.events)}\n")

        for i, ev in enumerate(self.events, 1):
            ts = ev.get("time")
            ts_str = ts.strftime("%Y-%m-%d %H:%M:%S") if ts else "Unknown time"
            print(f"  [{ts_str}]")
            print(f"    Vendor:  {ev.get('vendor_name', 'N/A')} (VID: {ev.get('vid', '?')})")
            print(f"    Product: {ev.get('product', 'N/A')} (PID: {ev.get('pid', '?')})")
            if ev.get("serial"):
                print(f"    Serial:  {ev['serial']}")
            print()


if __name__ == "__main__":
    parser = LinuxUSBLogParser()

    # Try dmesg first
    print("[*] Parsing kernel messages (dmesg)...")
    dmesg_events = parser.parse_dmesg()

    # Try syslog
    syslog_path = "/var/log/syslog"
    if os.path.exists(syslog_path):
        print(f"[*] Parsing {syslog_path}...")
        parser.parse_log_file(syslog_path)
    else:
        print(f"[*] {syslog_path} not found; trying kern.log...")
        kern_log = "/var/log/kern.log"
        if os.path.exists(kern_log):
            parser.parse_log_file(kern_log)

    parser.report()

Step 3: Cross-Platform USB Inspector

Combine registry and log parsers into a unified tool that works across environments and produces a forensic timeline.

#!/usr/bin/env python3

import argparse
import json
import csv
from datetime import datetime

class USBInspector:
    """Cross-platform USB device history inspector."""

    def __init__(self):
        self.devices = []
        self.linux_parser = LinuxUSBLogParser()

    def inspect_windows_registry(self, reg_path):
        """Inspect exported Windows SYSTEM registry hive."""
        parser = RegistryUSBParser()
        devices = parser.parse_reg_file(reg_path)
        for d in devices:
            d["source"] = "windows_registry"
            d["discovery_time"] = datetime.now().isoformat()
        self.devices.extend(devices)
        return devices

    def inspect_linux_logs(self, log_paths=None):
        """Inspect Linux system logs."""
        if log_paths:
            for lp in log_paths:
                self.linux_parser.parse_log_file(lp)
        else:
            self.linux_parser.parse_dmesg()
            for candidate in ["/var/log/syslog", "/var/log/kern.log", "/var/log/messages"]:
                if os.path.exists(candidate):
                    self.linux_parser.parse_log_file(candidate)

        for ev in self.linux_parser.events:
            device = {
                "vid": ev.get("vid", ""),
                "pid": ev.get("pid", ""),
                "vendor": ev.get("vendor_name", ""),
                "product": ev.get("product", ""),
                "serial": ev.get("serial", ""),
                "first_seen": ev.get("time").isoformat() if ev.get("time") else None,
                "source": "linux_logs",
            }
            self.devices.append(device)

        return self.linux_parser.events

    def export_json(self, output_path):
        with open(output_path, "w") as f:
            json.dump(self.devices, f, indent=2, default=str)
        print(f"[+] Exported {len(self.devices)} devices to {output_path}")

    def export_csv(self, output_path):
        if not self.devices:
            return
        fieldnames = self.devices[0].keys()
        with open(output_path, "w", newline="") as f:
            writer = csv.DictWriter(f, fieldnames=fieldnames)
            writer.writeheader()
            writer.writerows(self.devices)
        print(f"[+] Exported CSV to {output_path}")


if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="USB Device History Inspector")
    parser.add_argument("--reg", help="Exported Windows .reg file to analyze")
    parser.add_argument("--logs", nargs="*", help="Linux log file(s) to analyze")
    parser.add_argument("-o", "--output", help="JSON report output path")
    parser.add_argument("--csv", help="CSV report output path")
    args = parser.parse_args()

    inspector = USBInspector()

    if args.reg:
        print(f"[*] Analyzing Windows registry: {args.reg}")
        inspector.inspect_windows_registry(args.reg)

    if args.logs:
        print(f"[*] Analyzing Linux logs: {args.logs}")
        inspector.inspect_linux_logs(args.logs)
    elif not args.reg:
        # Default: inspect current Linux system
        print("[*] No input specified — inspecting current Linux system logs")
        inspector.inspect_linux_logs()

    if args.output:
        inspector.export_json(args.output)
    if args.csv:
        inspector.export_csv(args.csv)

    print(f"\n[*] Total USB devices found: {len(inspector.devices)}")

Testing Your Inspector

  • Windows: Export the USBSTOR key: reg export HKLM\SYSTEM\CurrentControlSet\Enum\USBSTOR usbstor.reg
  • Linux (live): Run sudo dmesg | grep -i usb then run your tool — compare results
  • Linux (log files): Analyze /var/log/syslog and /var/log/kern.log
  • Forensic images: Mount a forensic disk image and point the tool at the extracted registry hives
  • Compare with USB Detective: Run the commercial USB Detective tool on the same system to validate findings
