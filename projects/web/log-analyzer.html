Log Analyzer & Intrusion Detector

Parse web server logs to detect attack patterns — SQLi, XSS, path traversal, brute force — and generate alert reports.

  • Implement time-window analysis to detect brute force and credential stuffing
  • Build an alerting pipeline with severity classification and report generation

Step 1: Parse Apache/Nginx Access Logs

Start by parsing the Combined Log Format used by Apache and Nginx. Named capture groups produce clean dictionary output.

import re
from datetime import datetime

class LogParser:
    COMBINED_PATTERN = re.compile(
        r'^(?P<ip>\S+) \S+ \S+ '
        r'\[(?P<timestamp>[^\]]+)\] '
        r'"(?P<method>\S+) (?P<path>\S+) (?P<protocol>[^"]+)" '
        r'(?P<status>\d{3}) (?P<size>\S+) '
        r'"(?P<referer>[^"]*)" '
        r'"(?P<user_agent>[^"]*)"'
    )

    def parse_line(self, line):
        match = self.COMBINED_PATTERN.match(line.strip())
        if not match:
            return None
        event = match.groupdict()
        event['status'] = int(event['status'])
        event['size'] = int(event['size']) if event['size'].isdigit() else 0
        try:
            event['timestamp'] = datetime.strptime(
                event['timestamp'], '%d/%b/%Y:%H:%M:%S %z'
            )
        except ValueError:
            event['timestamp'] = datetime.now()
        if '?' in event['path']:
            event['url'], event['query_string'] = event['path'].split('?', 1)
        else:
            event['url'] = event['path']
            event['query_string'] = ''
        return event

parser = LogParser()
sample = '192.168.1.1 - - [01/Jun/2025:12:34:56 +0000] "GET /login?id=1 HTTP/1.1" 200 1234 "-" "Mozilla/5.0"'
print(parser.parse_line(sample))

Key concept: Split the path on ? to isolate the query string — that's where most injection payloads hide.

Step 2: Detect Web Attack Patterns

Write regex signatures for common web attacks. Each signature targets patterns in URLs, query strings, and user agents.

class AttackDetector:
    def __init__(self):
        self.signatures = {
            'sqli': [
                r"(?i)(\bUNION\s+SELECT\b)",
                r"(?i)(\bOR\s+['\"]?\d['\"]?\s*=\s*['\"]?\d['\"]?)",
                r"(?i)(\bSLEEP\s*\(.*\))",
                r"(?i)(--\s*[^a-zA-Z0-9]*$)",
                r"(?i)(\bWAITFOR\s+DELAY\b)",
            ],
            'xss': [
                r"(?i)(<script[^>]*>)",
                r"(?i)(javascript\s*:.*)",
                r"(?i)(on\w+\s*=\s*['\"].*['\"])",
                r"(?i)(<img[^>]+onerror\s*=)",
            ],
            'path_traversal': [
                r"(\.\.\/)+",
                r"(\.\.%2F)+",
                r"(%2F\.\.){2,}",
                r"(/etc/passwd|/etc/shadow)",
                r"(\bphp://filter\b|\bexpect://\b)",
            ],
            'cmd_injection': [
                r"(?i)(;\s*(id|whoami|ls|pwd|uname|cat|wget)\b)",
                r"(?i)(\|{1,2}\s*(id|whoami|ls|pwd))",
                r"(?i)(`[^`]+`)",
            ],
            'scanner_ua': [
                r"(?i)(sqlmap|nikto|nmap|acunetix|nessus|openvas|burpsuite|dirbuster|wfuzz)",
            ],
        }

    def analyze(self, event):
        detections = []
        haystack = f"{event.get('url','')} {event.get('query_string','')} {event.get('user_agent','')}"
        for category, patterns in self.signatures.items():
            for pattern in patterns:
                if re.search(pattern, haystack):
                    detections.append({
                        'category': category,
                        'ip': event.get('ip'),
                        'timestamp': event.get('timestamp'),
                        'path': event.get('path'),
                    })
                    break
        return detections

Pattern tip: (?i) makes matching case-insensitive. Start with precise patterns (known tool signatures) then add broader heuristics — loose patterns create false positives.

Step 3: Brute Force Detection via Time Windows

Not all attacks have obvious signatures. Detect brute force by tracking request frequency per IP over a sliding time window.

from collections import defaultdict
from datetime import datetime, timedelta

class RateBasedDetector:
    def __init__(self, window_seconds=300, threshold=10):
        self.window = timedelta(seconds=window_seconds)
        self.threshold = threshold
        self.attempts = defaultdict(list)

    def feed(self, event):
        ip = event.get('ip')
        ts = event.get('timestamp')
        path = event.get('url', '')
        if 'login' not in path.lower() and event.get('method') != 'POST':
            return None
        self.attempts[ip].append(ts)
        cutoff = ts - self.window
        self.attempts[ip] = [t for t in self.attempts[ip] if t > cutoff]
        count = len(self.attempts[ip])
        if count >= self.threshold:
            return {
                'type': 'brute_force',
                'ip': ip,
                'count': count,
                'window_seconds': self.window.total_seconds(),
                'first_seen': self.attempts[ip][0],
                'last_seen': ts,
                'severity': 'high' if count >= self.threshold * 2 else 'medium',
            }
        return None

This sliding window prunes old entries automatically — you only alert when the current rate exceeds the threshold, no stale alerts from hours ago.

Step 4: Generate Alert Reports

Build a reporter that summarizes findings with severity levels, top attackers, and actionable context.

import json
from collections import Counter, defaultdict

class AlertReporter:
    SEVERITY_MAP = {
        'sqli': 'critical', 'cmd_injection': 'critical',
        'lfi_rfi': 'high', 'xss': 'high',
        'path_traversal': 'medium', 'scanner_ua': 'low',
        'rate_limit': 'high',
    }

    def generate(self, detections, rate_alerts, total_events, time_range):
        report = {
            'generated_at': datetime.now().isoformat(),
            'time_range': time_range,
            'total_events': total_events,
            'summary': {},
            'alerts': [],
        }
        cats = defaultdict(lambda: {'count': 0, 'ips': set(), 'severity': 'low'})
        for d in detections:
            c = d['category']
            cats[c]['count'] += 1
            cats[c]['ips'].add(d['ip'])
            cats[c]['severity'] = self.SEVERITY_MAP.get(c, 'low')

        report['summary'] = {
            cat: {'count': d['count'], 'unique_ips': len(d['ips']), 'severity': d['severity']}
            for cat, d in sorted(cats.items(), key=lambda x: x[1]['count'], reverse=True)
        }

        for a in rate_alerts:
            report['alerts'].append({
                'type': 'brute_force', 'ip': a['ip'],
                'attempts': a['count'], 'severity': a['severity'],
                'message': f"{a['ip']} made {a['count']} login attempts in {a['window_seconds']}s",
            })

        ip_counter = Counter(d['ip'] for d in detections)
        report['top_attackers'] = [{'ip': ip, 'count': c} for ip, c in ip_counter.most_common(10)]
        return report

    def print_report(self, report):
        print("\\n" + "=" * 60)
        print(f"  LOG ANALYSIS REPORT — {report['generated_at'][:19]}")
        print("=" * 60)
        print(f"  Events analyzed: {report['total_events']}")
        if report['summary']:
            for cat, d in report['summary'].items():
                icon = {'critical': '🔴', 'high': '🟠', 'medium': '🟡', 'low': '🔵'}
                print(f"  {icon.get(d['severity'], '⚪')} {cat.upper()}: {d['count']} hits from {d['unique_ips']} IPs [{d['severity']}]")
        if report['top_attackers']:
            print(f"\\n  Top attacking IPs:")
            for e in report['top_attackers'][:5]:
                print(f"     {e['ip']:20s} — {e['count']} events")
        print("=" * 60)

Step 5: Complete CLI Tool

Wire everything together with argparse for a production-ready command-line tool.

import sys, argparse

def main():
    parser = argparse.ArgumentParser(description="Web Log Analyzer & Intrusion Detector")
    parser.add_argument("logfile", help="Path to Apache/Nginx access log")
    parser.add_argument("-o", "--output", help="Output JSON report file")
    parser.add_argument("-w", "--window", type=int, default=300, help="Rate window in seconds")
    parser.add_argument("-t", "--threshold", type=int, default=10, help="Rate threshold")
    parser.add_argument("--json", action="store_true", help="Output JSON")
    args = parser.parse_args()

    log_parser = LogParser()
    attack_detector = AttackDetector()
    rate_detector = RateBasedDetector(args.window, args.threshold)
    reporter = AlertReporter()

    all_detections, rate_alerts, events = [], [], []
    with open(args.logfile, 'r', encoding='utf-8', errors='ignore') as f:
        for line in f:
            event = log_parser.parse_line(line)
            if not event:
                continue
            events.append(event)
            all_detections.extend(attack_detector.analyze(event))
            alert = rate_detector.feed(event)
            if alert:
                rate_alerts.append(alert)

    report = reporter.generate(all_detections, rate_alerts, len(events),
                               f"{events[0]['timestamp']} → {events[-1]['timestamp']}")
    if args.json and args.output:
        with open(args.output, 'w') as f:
            json.dump(report, f, default=str, indent=2)
        print(f"Report saved to {args.output}")
    else:
        reporter.print_report(report)

if __name__ == "__main__":
    main()
