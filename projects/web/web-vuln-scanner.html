Web Vulnerability Scanner

Build an automated scanner that checks for common web vulnerabilities — SQLi, XSS, CSRF, misconfigurations.

pythonhttpweb-securityowasp

Learning Objectives

  • Send and parse HTTP requests/responses programmatically with the requests library
  • Check for missing or misconfigured security headers (HSTS, CSP, X-Frame-Options, etc.)
  • Implement automated SQL injection and XSS payload injection
  • Detect common misconfigurations like directory listing and exposed .git folders
  • Generate a structured vulnerability report with severity ratings

Prerequisites

  • Python 3.8+ with requests and beautifulsoup4 installed
  • Understanding of HTTP methods, headers, status codes
  • Familiarity with OWASP Top 10 vulnerability categories
  • A deliberately vulnerable test target (DVWA, OWASP Juice Shop, or your own app)
⚠ Ethics & Legal

Only scan websites you own or have explicit written permission to test. Unauthorized vulnerability scanning is illegal and can trigger intrusion detection systems. Always test against local lab environments like DVWA or OWASP Juice Shop.

Step 1: Security Header Audit

Start with passive checks — security headers are the easiest to verify and often the most neglected. Missing headers are a reliable signal of poor security posture.

import requests

REQUIRED_HEADERS = {
    'Strict-Transport-Security': {
        'severity': 'high',
        'desc': 'Enforces HTTPS. Missing = MITM risk.',
        'recommendation': 'max-age=31536000; includeSubDomains'
    },
    'Content-Security-Policy': {
        'severity': 'medium',
        'desc': 'Prevents XSS and data injection.',
        'recommendation': "default-src 'self'"
    },
    'X-Frame-Options': {
        'severity': 'medium',
        'desc': 'Prevents clickjacking.',
        'recommendation': 'DENY or SAMEORIGIN'
    },
    'X-Content-Type-Options': {
        'severity': 'low',
        'desc': 'Prevents MIME sniffing.',
        'recommendation': 'nosniff'
    },
    'Referrer-Policy': {
        'severity': 'low',
        'desc': 'Controls referrer information leakage.',
        'recommendation': 'strict-origin-when-cross-origin'
    },
    'Permissions-Policy': {
        'severity': 'low',
        'desc': 'Restricts browser features (camera, mic, etc.).',
        'recommendation': 'camera=(), microphone=(), geolocation=()'
    },
}

def check_security_headers(url):
    findings = []
    try:
        resp = requests.get(url, timeout=10, allow_redirects=True)
        for header, info in REQUIRED_HEADERS.items():
            if header not in resp.headers:
                findings.append({
                    'type': 'missing_header',
                    'header': header,
                    'severity': info['severity'],
                    'description': info['desc'],
                    'recommendation': info['recommendation'],
                })
        return findings, resp.headers
    except requests.RequestException as e:
        return [{'type': 'connection_error', 'error': str(e)}], {}

findings, headers = check_security_headers('http://localhost:8080')
for f in findings:
    print(f"[{f.get('severity','error').upper()}] {f.get('header', f.get('error',''))}")
    if 'recommendation' in f:
        print(f"  Fix: {f['recommendation']}")

Step 2: Common Misconfiguration Checks

Scan for exposed sensitive paths and misconfigurations that leak information or grant unintended access.

import concurrent.futures

MISCONFIG_CHECKS = {
    '/.git/HEAD': 'Exposed Git repository',
    '/.env': 'Exposed environment file',
    '/.DS_Store': 'macOS metadata file',
    '/wp-config.php.bak': 'WordPress config backup',
    '/backup/': 'Open backup directory',
    '/phpinfo.php': 'PHP info disclosure',
    '/server-status': 'Apache server status',
    '/.htaccess': 'Apache config exposed',
    '/admin/': 'Admin panel accessible',
    '/console/': 'H2/Spring console',
}

def check_misconfigs(base_url, max_workers=10):
    findings = []
    def check_one(path, description):
        url = base_url.rstrip('/') + path
        try:
            r = requests.get(url, timeout=5, allow_redirects=False)
            if r.status_code in (200, 403):
                findings.append({
                    'type': 'misconfig',
                    'url': url,
                    'status': r.status_code,
                    'description': description,
                    'severity': 'high' if r.status_code == 200 else 'medium',
                })
        except:
            pass

    with concurrent.futures.ThreadPoolExecutor(max_workers=max_workers) as ex:
        futures = [ex.submit(check_one, path, desc) for path, desc in MISCONFIG_CHECKS.items()]
        concurrent.futures.wait(futures)

    return findings

Step 3: Automated XSS Probe

Detect reflected XSS by injecting benign payloads into query parameters and checking if they appear unescaped in the response.

from urllib.parse import urlparse, parse_qs, urlencode, urlunparse

XSS_PAYLOADS = [
    '',
    '">',
    '',
    '\' onfocus=alert(1) autofocus>',
    'javascript:alert(1)',
]

def probe_xss(url):
    """Test GET parameters for reflected XSS."""
    parsed = urlparse(url)
    params = parse_qs(parsed.query, keep_blank_values=True)
    findings = []

    if not params:
        return findings

    for param_name in params:
        original = params[param_name][0] if params[param_name] else ''
        for payload in XSS_PAYLOADS:
            test_params = dict(params)
            test_params[param_name] = [payload]
            new_query = urlencode(test_params, doseq=True)
            test_url = urlunparse(parsed._replace(query=new_query))

            try:
                r = requests.get(test_url, timeout=5)
                if payload in r.text and 'Content-Type' in r.headers:
                    ct = r.headers['Content-Type']
                    if 'text/html' in ct:
                        findings.append({
                            'type': 'reflected_xss',
                            'parameter': param_name,
                            'payload': payload,
                            'severity': 'high',
                            'url': test_url,
                        })
                        break  # One payload match per param is enough
            except:
                continue

    return findings

Step 4: SQL Injection Detection

Probe parameters with error-inducing payloads and analyze responses for database error messages.

SQLI_ERRORS = [
    'SQL syntax', 'mysql_fetch', 'ORA-', 'PostgreSQL',
    'SQLite', 'unclosed quotation mark', 'syntax error',
    'Microsoft OLE DB', 'ODBC Driver', 'SQL command',
    'Unknown column', 'Incorrect syntax',
]

SQLI_PROBES = [
    "'",          # Break string
    "\"",         # Break double-quoted string
    "' OR '1'='1",  # Tautology
    "' OR 1=1--",  # Tautology with comment
    "1' AND SLEEP(5)--",  # Time-based (MySQL)
    "1' AND '1'='1",  # Stacked
]

def probe_sqli(url):
    parsed = urlparse(url)
    params = parse_qs(parsed.query, keep_blank_values=True)
    findings = []
    if not params:
        return findings

    baseline = requests.get(url, timeout=5)

    for param_name in params:
        for probe in SQLI_PROBES:
            test_params = dict(params)
            test_params[param_name] = [probe]
            new_query = urlencode(test_params, doseq=True)
            test_url = urlunparse(parsed._replace(query=new_query))

            try:
                r = requests.get(test_url, timeout=10)
                body = r.text.lower()

                # Check for DB error messages
                for err_sig in SQLI_ERRORS:
                    if err_sig.lower() in body:
                        findings.append({
                            'type': 'sqli_error_based',
                            'parameter': param_name,
                            'payload': probe,
                            'error_signature': err_sig,
                            'severity': 'critical',
                            'url': test_url,
                        })
                        return findings  # One confirmed finding per param

                # Check for time-based (response significantly slower)
                if 'SLEEP' in probe.upper() and r.elapsed.total_seconds() > 4:
                    findings.append({
                        'type': 'sqli_time_based',
                        'parameter': param_name,
                        'payload': probe,
                        'response_time': r.elapsed.total_seconds(),
                        'severity': 'critical',
                        'url': test_url,
                    })
                    return findings

            except requests.Timeout:
                # Timeout >5s confirms time-based
                if 'SLEEP' in probe.upper():
                    findings.append({
                        'type': 'sqli_time_based',
                        'parameter': param_name,
                        'payload': probe,
                        'severity': 'critical',
                    })
                    return findings
            except:
                continue

    return findings

Step 5: Complete Scanner CLI

Assemble all modules into a single vulnerability scanner with arg-parsing and a formatted report.

import argparse, json
from datetime import datetime

def main():
    parser = argparse.ArgumentParser(description="Web Vulnerability Scanner")
    parser.add_argument("url", help="Target URL")
    parser.add_argument("--headers-only", action="store_true", help="Only check headers")
    parser.add_argument("--no-xss", action="store_true", help="Skip XSS probing")
    parser.add_argument("--no-sqli", action="store_true", help="Skip SQLi probing")
    parser.add_argument("-o", "--output", help="Save JSON report")
    parser.add_argument("-t", "--threads", type=int, default=10, help="Thread count")
    args = parser.parse_args()

    url = args.url if args.url.startswith('http') else f'http://{args.url}'
    all_findings = []

    # 1. Headers
    print("[*] Checking security headers...")
    header_findings, _ = check_security_headers(url)
    all_findings.extend(header_findings)
    print(f"  -> {len(header_findings)} header issues found")

    # 2. Misconfigs
    print("[*] Checking common misconfigurations...")
    misconfig_findings = check_misconfigs(url, args.threads)
    all_findings.extend(misconfig_findings)
    print(f"  -> {len(misconfig_findings)} misconfigs found")

    # 3. XSS
    if not args.no_xss and not args.headers_only:
        print("[*] Probing for reflected XSS...")
        xss_findings = probe_xss(url)
        all_findings.extend(xss_findings)
        print(f"  -> {len(xss_findings)} XSS findings")

    # 4. SQLi
    if not args.no_sqli and not args.headers_only:
        print("[*] Probing for SQL injection...")
        sqli_findings = probe_sqli(url)
        all_findings.extend(sqli_findings)
        print(f"  -> {len(sqli_findings)} SQLi findings")

    # Report
    report = {
        'scan_time': datetime.now().isoformat(),
        'target': url,
        'total_findings': len(all_findings),
        'severity_counts': {
            'critical': sum(1 for f in all_findings if f.get('severity') == 'critical'),
            'high': sum(1 for f in all_findings if f.get('severity') == 'high'),
            'medium': sum(1 for f in all_findings if f.get('severity') == 'medium'),
            'low': sum(1 for f in all_findings if f.get('severity') == 'low'),
        },
        'findings': all_findings,
    }

    print(f"\n{'='*60}")
    print(f"  SCAN COMPLETE — {report['total_findings']} findings")
    print(f"  Critical: {report['severity_counts']['critical']} | "
          f"High: {report['severity_counts']['high']} | "
          f"Medium: {report['severity_counts']['medium']} | "
          f"Low: {report['severity_counts']['low']}")
    print(f"{'='*60}")

    if args.output:
        with open(args.output, 'w') as f:
            json.dump(report, f, indent=2)
        print(f"\nReport saved to {args.output}")

if __name__ == "__main__":
    main()
