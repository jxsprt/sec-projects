Directory Brute-Forcer

Build a tool to discover hidden directories and files on web servers using wordlist-based brute force.

pythonhttpreconenumeration

Learning Objectives

  • Understand HTTP status codes (200, 301, 403, 404) and what they reveal about directory structure
  • Build a multi-threaded HTTP client for concurrent directory enumeration
  • Implement response analysis to reduce false positives (custom 404 pages, redirects)
  • Work with wordlists — load, filter, and generate directory paths
  • Add recursive discovery to map entire directory trees

Prerequisites

  • Python 3.8+ with requests installed
  • Basic HTTP knowledge (status codes, headers, URLs)
  • A wordlist file (SecLists recommended) or we'll generate a basic one
⚠ Ethics & Legal

Only brute-force directories on websites you own or have permission to test. Aggressive scanning generates noise in server logs and can trigger rate limiting or IP bans. Always respect robots.txt and add configurable delays between requests.

Step 1: Basic Single-Threaded Brute-Forcer

Start with a simple loop that tries each wordlist entry against the target. This establishes the core logic before we optimize with threading.

import requests
import sys

class DirBruteForcer:
    def __init__(self, base_url, wordlist_path, extensions=None):
        self.base_url = base_url.rstrip('/')
        self.extensions = extensions or ['']  # '' = no extension
        self.session = requests.Session()
        self.session.headers['User-Agent'] = (
            'Mozilla/5.0 (compatible; DirBruteForcer/1.0; educational)'
        )
        self.results = []
        self.baseline = None  # For false positive detection

    def load_wordlist(self, path):
        with open(path, 'r', encoding='utf-8', errors='ignore') as f:
            return [line.strip() for line in f if line.strip()]

    def get_baseline(self):
        """Request a non-existent path to learn how 404s look."""
        fake_url = f"{self.base_url}/_nonexistent_path_12345_"
        try:
            self.baseline = self.session.get(fake_url, timeout=10)
        except:
            self.baseline = None

    def scan(self, wordlist_path):
        words = self.load_wordlist(wordlist_path)
        self.get_baseline()

        for word in words:
            for ext in self.extensions:
                path = f"/{word}{ext}"
                url = f"{self.base_url}{path}"
                try:
                    r = self.session.get(url, timeout=5, allow_redirects=False)
                    if self._is_interesting(r, url):
                        self.results.append({
                            'url': url,
                            'status': r.status_code,
                            'content_length': len(r.content),
                            'redirect_to': r.headers.get('Location', ''),
                        })
                        print(f"[{r.status_code}] {url} ({len(r.content)} bytes)")
                except requests.RequestException:
                    continue

        return self.results

    def _is_interesting(self, response, url):
        """Filter out false positives."""
        # Actual 404
        if response.status_code == 404:
            return False
        # Redirects might be interesting
        if response.status_code in (301, 302, 307, 308):
            return True
        # 200 OK - but check it's not a custom 404 page
        if response.status_code == 200:
            if self.baseline and response.status_code == self.baseline.status_code:
                if abs(len(response.content) - len(self.baseline.content)) < 100:
                    return False  # Same as 404 = custom error page
            return True
        # 401/403 = exists but access denied (interesting!)
        if response.status_code in (401, 403):
            return True
        # 500 = might be interesting
        if response.status_code in (500, 503):
            return True
        return False

The baseline trick: By requesting a guaranteed non-existent URL first, we learn how the server behaves on 404s. Many sites return 200 with a "page not found" body — comparing content length helps filter them out.

Step 2: Multi-Threaded Speed Boost

Single-threaded scanning is slow. Use ThreadPoolExecutor to fire concurrent requests — but be responsible with rate limiting.

import concurrent.futures
import time

class ThreadedDirBruteForcer(DirBruteForcer):
    def __init__(self, base_url, wordlist_path, extensions=None,
                 threads=20, delay=0):
        super().__init__(base_url, wordlist_path, extensions)
        self.threads = threads
        self.delay = delay  # seconds between requests per thread

    def scan(self, wordlist_path):
        words = self.load_wordlist(wordlist_path)
        self.get_baseline()

        # Build all URLs to test
        urls_to_check = []
        for word in words:
            for ext in self.extensions:
                path = f"/{word}{ext}"
                urls_to_check.append({
                    'url': f"{self.base_url}{path}",
                    'word': word,
                    'ext': ext,
                })

        print(f"\n[*] Scanning {len(urls_to_check)} paths with {self.threads} threads...\n")

        with concurrent.futures.ThreadPoolExecutor(max_workers=self.threads) as executor:
            futures = {executor.submit(self._check_url, entry): entry
                      for entry in urls_to_check}

            for future in concurrent.futures.as_completed(futures):
                result = future.result()
                if result:
                    self.results.append(result)
                    print(f"[{result['status']}] {result['url']} "
                          f"({result['content_length']} bytes)")

        self.results.sort(key=lambda x: x['status'])
        return self.results

    def _check_url(self, entry):
        if self.delay:
            time.sleep(self.delay)
        try:
            r = self.session.get(entry['url'], timeout=5,
                                allow_redirects=False)
            if self._is_interesting(r, entry['url']):
                return {
                    'url': entry['url'],
                    'status': r.status_code,
                    'content_length': len(r.content),
                    'redirect_to': r.headers.get('Location', ''),
                    'word': entry['word'],
                }
        except:
            pass
        return None

Step 3: Recursive Discovery & Common Extensions

When you find a directory, recurse into it to build a full site map. Add extension fuzzing for backup files and common development artifacts.

BACKUP_EXTENSIONS = ['.bak', '.old', '.save', '.swp', '~', '.orig']
DEV_EXTENSIONS = ['.php', '.asp', '.aspx', '.jsp', '.py', '.rb']
CONFIG_FILES = ['.htaccess', '.gitignore', 'web.config', 'robots.txt']

class RecursiveDirBruteForcer(ThreadedDirBruteForcer):
    def __init__(self, *args, max_depth=3, **kwargs):
        super().__init__(*args, **kwargs)
        self.max_depth = max_depth
        self.visited = set()

    def scan_recursive(self, wordlist_path, current_path='', depth=0):
        if depth > self.max_depth:
            return []

        all_results = []
        words = self.load_wordlist(wordlist_path)
        base = f"{self.base_url}{current_path}"

        # Check config files at each level
        for config in CONFIG_FILES:
            url = f"{base}/{config}"
            try:
                r = self.session.get(url, timeout=5)
                if r.status_code in (200, 403):
                    all_results.append({
                        'url': url, 'status': r.status_code,
                        'type': 'config_file', 'content_length': len(r.content),
                    })
            except:
                pass

        urls_to_check = []
        for word in words:
            for ext in self.extensions:
                path = f"{current_path}/{word}{ext}"
                urls_to_check.append({'url': f"{self.base_url}{path}", 'word': word, 'ext': ext})

        with concurrent.futures.ThreadPoolExecutor(max_workers=self.threads) as executor:
            futures = {executor.submit(self._check_url, e): e for e in urls_to_check}
            for future in concurrent.futures.as_completed(futures):
                result = future.result()
                if result:
                    result['depth'] = depth
                    all_results.append(result)
                    print(f"[{result['status']}] {result['url']}")

                    # Recurse into directories
                    if result['status'] in (200, 301, 302, 403) and '/' in result['url']:
                        dir_path = result['url'].replace(self.base_url, '')
                        if ('/' in dir_path[len(current_path)+1:] and
                            dir_path not in self.visited):
                            self.visited.add(dir_path)

        return all_results

Step 4: CLI with Reporting

Tie it together with a clean CLI and JSON report output.

import argparse, json
from datetime import datetime

def main():
    parser = argparse.ArgumentParser(description="Directory Brute-Forcer")
    parser.add_argument("url", help="Target URL")
    parser.add_argument("-w", "--wordlist", required=True, help="Path to wordlist file")
    parser.add_argument("-e", "--extensions", default="",
                        help="Comma-separated extensions (e.g., php,txt,bak)")
    parser.add_argument("-t", "--threads", type=int, default=20,
                        help="Number of threads (default: 20)")
    parser.add_argument("-d", "--delay", type=float, default=0,
                        help="Delay between requests in seconds")
    parser.add_argument("-r", "--recursive", action="store_true",
                        help="Enable recursive scanning")
    parser.add_argument("--max-depth", type=int, default=3,
                        help="Max recursion depth (default: 3)")
    parser.add_argument("-o", "--output", help="Save JSON report")
    args = parser.parse_args()

    extensions = args.extensions.split(',') if args.extensions else ['']

    if args.recursive:
        scanner = RecursiveDirBruteForcer(
            args.url, args.wordlist, extensions,
            threads=args.threads, delay=args.delay,
            max_depth=args.max_depth
        )
        results = scanner.scan_recursive(args.wordlist)
    else:
        scanner = ThreadedDirBruteForcer(
            args.url, args.wordlist, extensions,
            threads=args.threads, delay=args.delay
        )
        results = scanner.scan(args.wordlist)

    # Summary
    status_counts = {}
    for r in results:
        status_counts[r['status']] = status_counts.get(r['status'], 0) + 1

    print(f"\\n{'='*60}")
    print(f"  SCAN COMPLETE — {len(results)} paths found")
    for status, count in sorted(status_counts.items()):
        print(f"  HTTP {status}: {count}")
    print(f"{'='*60}")

    if args.output:
        report = {
            'scan_time': datetime.now().isoformat(),
            'target': args.url,
            'total_found': len(results),
            'status_counts': status_counts,
            'results': results,
        }
        with open(args.output, 'w') as f:
            json.dump(report, f, indent=2)
        print(f"\\nReport saved to {args.output}")

if __name__ == "__main__":
    main()
