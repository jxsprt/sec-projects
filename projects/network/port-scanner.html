Port Scanner

⚠ Ethics & Legal

Only scan systems you own or have explicit written permission to test. Unauthorized port scanning is illegal in most jurisdictions and can be detected by intrusion detection systems.

Step 1: Basic Single-Port Scanner

Start with a simple function that attempts a TCP connection to a single port. If the connection succeeds, the port is open.

import socket
import sys

def scan_port(host, port):
    """Attempt TCP connection to host:port. Returns True if open."""
    try:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(1.0)  # 1 second timeout
        result = sock.connect_ex((host, port))
        sock.close()
        return result == 0  # 0 = success, port open
    except socket.gaierror:
        print(f"Error: Could not resolve hostname '{host}'")
        sys.exit(1)
    except socket.error as e:
        print(f"Socket error: {e}")
        return False

# Test it
host = "scanme.nmap.org"  # Nmap's test server — safe to scan!
port = 80
if scan_port(host, port):
    print(f"Port {port} on {host} is OPEN")
else:
    print(f"Port {port} on {host} is CLOSED")

Key concepts: socket.AF_INET for IPv4, socket.SOCK_STREAM for TCP, connect_ex() returns 0 on success (unlike connect() which raises an exception).

Step 2: Scan a Range of Ports

Extend the scanner to check multiple ports. We'll use common ports (1-1024) plus some well-known higher ports.

def scan_port_range(host, start_port, end_port):
    """Scan a range of ports on a host."""
    open_ports = []

    for port in range(start_port, end_port + 1):
        if scan_port(host, port):
            open_ports.append(port)
            print(f"[+] Port {port} is OPEN")

    return open_ports

# Scan common ports
host = "scanme.nmap.org"
open_ports = scan_port_range(host, 1, 1024)
print(f"\nOpen ports: {open_ports}")

Step 3: Add Threading for Speed

Serial scanning is slow — each closed port wastes ~1 second on timeout. Use concurrent.futures to scan in parallel.

import concurrent.futures

def fast_scan(host, ports, max_workers=100):
    """Scan ports concurrently using thread pool."""
    open_ports = []

    with concurrent.futures.ThreadPoolExecutor(max_workers=max_workers) as executor:
        future_to_port = {
            executor.submit(scan_port, host, port): port
            for port in ports
        }

        for future in concurrent.futures.as_completed(future_to_port):
            port = future_to_port[future]
            try:
                if future.result():
                    open_ports.append(port)
                    print(f"[+] Port {port} is OPEN")
            except Exception as e:
                print(f"[-] Error scanning port {port}: {e}")

    return sorted(open_ports)

# Scan top 1000 ports in parallel
common_ports = list(range(1, 1001))
open_ports = fast_scan("scanme.nmap.org", common_ports)

Step 4: Banner Grabbing (Service Detection)

Once a port is open, try to identify the service by reading the banner it sends.

COMMON_BANNER_PORTS = {
    21: b"220",   # FTP
    22: b"SSH",   # SSH
    25: b"220",   # SMTP
    80: b"GET / HTTP/1.0\r\n\r\n",  # HTTP
    443: b"",     # HTTPS (needs TLS)
}

def grab_banner(host, port, timeout=2.0):
    """Attempt to grab service banner from open port."""
    try:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(timeout)
        sock.connect((host, port))

        # Send probe if we know what to send
        probe = COMMON_BANNER_PORTS.get(port, b"")
        if probe:
            sock.send(probe)

        banner = sock.recv(1024)
        sock.close()
        return banner.decode('utf-8', errors='ignore').strip()
    except:
        return None

Step 5: Complete CLI Tool

Put it together with argparse for a professional command-line interface.

import argparse

def main():
    parser = argparse.ArgumentParser(
        description="TCP Port Scanner — fast, threaded, with banner grabbing"
    )
    parser.add_argument("host", help="Target hostname or IP")
    parser.add_argument("-p", "--ports", default="1-1000",
                        help="Port range (e.g., '1-1000' or '22,80,443')")
    parser.add_argument("-t", "--threads", type=int, default=100,
                        help="Number of threads (default: 100)")
    parser.add_argument("-b", "--banner", action="store_true",
                        help="Enable banner grabbing")
    parser.add_argument("--timeout", type=float, default=1.0,
                        help="Connection timeout in seconds")

    args = parser.parse_args()

    # Parse port specification
    if "," in args.ports:
        ports = [int(p) for p in args.ports.split(",")]
    elif "-" in args.ports:
        start, end = args.ports.split("-")
        ports = list(range(int(start), int(end) + 1))
    else:
        ports = [int(args.ports)]

    print(f"\nScanning {args.host} ({len(ports)} ports)...\n")

    open_ports = fast_scan(args.host, ports, args.threads)

    if args.banner:
        for port in sorted(open_ports):
            banner = grab_banner(args.host, port)
            if banner:
                print(f"  Port {port}: {banner[:80]}")

    print(f"\nDone. {len(open_ports)} open ports found.")

if __name__ == "__main__":
    main()
