Basic Hash Cracker

Build a dictionary-based hash cracker supporting MD5, SHA1, and SHA256 — understand how password cracking tools work under the hood.

Learning Objectives

  • Understand how one-way hash functions (MD5, SHA1, SHA256) are used in password storage
  • Implement dictionary attacks — the most common password cracking technique
  • Learn why salted hashes are essential (and why unsalted hashes are trivially crackable)
  • Compare cracking speed across different hash algorithms
  • Gain insight into how tools like Hashcat and John the Ripper work internally

Prerequisites

  • Python 3.8+ installed
  • Understanding of Python file I/O and string manipulation
  • Download a wordlist (we'll use the classic rockyou.txt)

Only crack hashes you own or have explicit permission to test. Hash cracking tools are legitimate for security auditing, password recovery, and education — but unauthorized use against systems you don't own is illegal.

Step 1: Hash Identification & Computation

First, build a tool to compute and identify hashes. Different algorithms produce different output lengths and patterns that let us detect the algorithm.

import hashlib
import re

def compute_hash(text, algorithm='sha256'):
    """Compute the hash of a string using the specified algorithm."""
    h = hashlib.new(algorithm)
    h.update(text.encode('utf-8'))
    return h.hexdigest()

def identify_hash(hash_str):
    """Identify likely hash algorithm from format."""
    hash_str = hash_str.strip().lower()
    length = len(hash_str)

    # MD5: 32 hex chars
    if length == 32 and re.match(r'^[a-f0-9]{32}$', hash_str):
        return ['md5', 'ntlm']
    # SHA1: 40 hex chars
    elif length == 40 and re.match(r'^[a-f0-9]{40}$', hash_str):
        return ['sha1']
    # SHA256: 64 hex chars
    elif length == 64 and re.match(r'^[a-f0-9]{64}$', hash_str):
        return ['sha256']
    # SHA512: 128 hex chars
    elif length == 128 and re.match(r'^[a-f0-9]{128}$', hash_str):
        return ['sha512']
    else:
        return ['unknown']

# Test
test_hashes = {
    'md5': compute_hash('password', 'md5'),
    'sha1': compute_hash('password', 'sha1'),
    'sha256': compute_hash('password', 'sha256'),
}
for algo, h in test_hashes.items():
    print(f"{algo.upper():6s}: {h} -> {identify_hash(h)}")

Step 2: Dictionary Attack Engine

The core attack: read a wordlist line by line, hash each candidate, and compare against target hashes. This is fast because we can hash once and compare against multiple targets.

import time

class HashCracker:
    def __init__(self):
        self.algorithms = ['md5', 'sha1', 'sha256']
        self.found = {}  # hash -> plaintext

    def dictionary_attack(self, target_hashes, wordlist_path, algorithm='sha256'):
        """
        Attempt to crack hashes using a wordlist.

        Args:
            target_hashes: set of hash strings to crack
            wordlist_path: path to wordlist file (one word per line)
            algorithm: hash algorithm to use

        Returns:
            dict of {hash: plaintext} for cracked hashes
        """
        target_hashes = {h.strip().lower() for h in target_hashes}
        cracked = {}
        start_time = time.time()
        attempts = 0

        try:
            with open(wordlist_path, 'r', encoding='utf-8', errors='ignore') as f:
                for line in f:
                    word = line.strip()
                    if not word:
                        continue

                    attempts += 1
                    h = compute_hash(word, algorithm)

                    if h in target_hashes:
                        cracked[h] = word
                        print(f"[+] CRACKED: {h} = '{word}'")
                        # Early exit if all cracked
                        if len(cracked) == len(target_hashes):
                            break

        except FileNotFoundError:
            print(f"Wordlist not found: {wordlist_path}")
            return cracked

        elapsed = time.time() - start_time
        speed = attempts / elapsed if elapsed > 0 else 0
        print(f"\nSearched {attempts:,} candidates in {elapsed:.1f}s ({speed:,.0f} hashes/sec)")
        print(f"Cracked: {len(cracked)}/{len(target_hashes)}")
        return cracked

# Example usage
cracker = HashCracker()
targets = {
    compute_hash('letmein', 'md5'),
    compute_hash('trustno1', 'md5'),
    compute_hash('admin123', 'md5'),
}
# cracker.dictionary_attack(targets, 'rockyou.txt', 'md5')

Performance note: Without a GPU, expect ~100K–500K hashes/sec for MD5, slower for SHA256. Hashcat on a GPU can do billions per second — it's worth testing this to understand the scale difference.

Step 3: Rule-Based Mutation (Beyond Plain Dictionary)

Plain dictionary attacks miss passwords like "Password1" or "admin2024!". Rule-based mutation applies common transformations to each dictionary word.

def apply_rules(word):
    """Generate common mutations of a word for rule-based attacks."""
    mutations = {word, word.lower(), word.upper(), word.capitalize()}

    # Common suffixes
    for suffix in ['1', '123', '1234', '12345', '!', '@', '#',
                   '2024', '2023', '2022', '2021', '2020']:
        mutations.add(word + suffix)

    # Leet speak substitutions
    leet = word.replace('a', '4').replace('e', '3').replace('i', '1')\
                .replace('o', '0').replace('s', '5').replace('t', '7')
    mutations.add(leet)

    # Reverse
    mutations.add(word[::-1])

    # Duplicate
    mutations.add(word + word)

    return mutations

def rule_based_attack(target_hashes, wordlist_path, algorithm='sha256', max_rules=10):
    """Dictionary attack with rule-based mutations."""
    target_set = {h.strip().lower() for h in target_hashes}
    cracked = {}
    attempts = 0
    rules_applied = 0

    with open(wordlist_path, 'r', encoding='utf-8', errors='ignore') as f:
        for line in f:
            word = line.strip()
            if not word:
                continue

            # Apply rules
            candidates = apply_rules(word)
            rules_applied += len(candidates)

            for candidate in candidates:
                attempts += 1
                h = compute_hash(candidate, algorithm)

                if h in target_set:
                    cracked[h] = candidate
                    print(f"[+] CRACKED: {h} = '{candidate}' (base: '{word}')")

                    if len(cracked) == len(target_set):
                        return cracked

    print(f"Total attempts: {attempts:,} (from {rules_applied:,} mutations)")
    return cracked

Why rules matter: Most password policies require digits or special characters. Users satisfy these with predictable patterns (word + "1", word + "!"). Rule-based attacks exploit this predictability — and they're much faster than pure brute-force.

Step 4: Complete CLI Cracker

Combine everything into a polished, multi-threaded CLI tool with progress reporting.

import argparse
from concurrent.futures import ThreadPoolExecutor
import sys

def main():
    parser = argparse.ArgumentParser(description='Basic Hash Cracker')
    parser.add_argument('-w', '--wordlist', required=True, help='Path to wordlist')
    parser.add_argument('-H', '--hash', action='append', help='Hash to crack (can repeat)')
    parser.add_argument('-f', '--hashfile', help='File with hashes (one per line)')
    parser.add_argument('-a', '--algorithm', default='sha256',
                        choices=['md5', 'sha1', 'sha256', 'sha512'],
                        help='Hash algorithm')
    parser.add_argument('-r', '--rules', action='store_true', help='Enable rule-based mutations')
    parser.add_argument('-o', '--output', help='Save cracked hashes to file')
    parser.add_argument('--identify', metavar='HASH', help='Identify hash type and exit')

    args = parser.parse_args()

    # Hash identification mode
    if args.identify:
        algs = identify_hash(args.identify)
        print(f"Hash: {args.identify}")
        print(f"Likely algorithms: {', '.join(algs)}")
        sys.exit(0)

    # Collect target hashes
    targets = set()
    if args.hash:
        targets.update(args.hash)
    if args.hashfile:
        with open(args.hashfile) as f:
            targets.update(line.strip() for line in f if line.strip())

    if not targets:
        parser.error("No hashes provided. Use -H or -f.")

    print(f"Target hashes: {len(targets)}")
    print(f"Algorithm: {args.algorithm}")
    print(f"Wordlist: {args.wordlist}")
    print(f"Rules: {'ON' if args.rules else 'OFF'}")
    print("=" * 50)

    cracker = HashCracker()

    if args.rules:
        results = rule_based_attack(targets, args.wordlist, args.algorithm)
    else:
        results = cracker.dictionary_attack(targets, args.wordlist, args.algorithm)

    # Output results
    if args.output:
        with open(args.output, 'w') as f:
            for h, plain in results.items():
                f.write(f"{h}:{plain}\n")
        print(f"\nResults saved to {args.output}")

    # Summary
    print(f"\n{'='*50}")
    print(f"CRACKED: {len(results)}/{len(targets)}")
    if results:
        print("\nResults:")
        for h, plain in results.items():
            print(f"  {h[:16]}... -> {plain}")

if __name__ == '__main__':
    main()
