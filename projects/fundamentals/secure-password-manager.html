Secure Password Manager

Develop a CLI password manager with encrypted storage, master password authentication, and password generation.

Learning Objectives

  • Build a secure password generator with entropy estimation
  • Use SQLite for encrypted credential storage — understand tradeoffs vs plaintext DBs

Step 1: Crypto Engine — Master Password & Vault Encryption

The crypto engine handles all encryption. The master password derives a key via PBKDF2, which encrypts the entire vault. We store a bcrypt hash to verify the master password without needing to decrypt.

from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
from cryptography.hazmat.primitives.ciphers import Cipher, algorithms, modes
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.backends import default_backend
from cryptography.exceptions import InvalidTag
import bcrypt
import os
import json
import base64
import secrets

class CryptoEngine:
    def __init__(self):
        self.backend = default_backend()

    def hash_master_password(self, password):
        """Hash for storage — used to verify master password later."""
        return bcrypt.hashpw(password.encode(), bcrypt.gensalt()).decode()

    def verify_master_password(self, stored_hash, password):
        return bcrypt.checkpw(password.encode(), stored_hash.encode())

    def derive_key(self, password, salt=None):
        if salt is None:
            salt = secrets.token_bytes(32)
        kdf = PBKDF2HMAC(
            algorithm=hashes.SHA256(), length=32, salt=salt,
            iterations=600_000, backend=self.backend
        )
        return kdf.derive(password.encode()), salt

    def encrypt_vault(self, data, master_password):
        """Encrypt the entire vault (JSON string) with AES-256-GCM."""
        key, salt = self.derive_key(master_password)
        nonce = secrets.token_bytes(12)

        cipher = Cipher(algorithms.AES(key), modes.GCM(nonce),
                        backend=self.backend)
        encryptor = cipher.encryptor()
        ciphertext = encryptor.update(data.encode()) + encryptor.finalize()

        return {
            'salt': base64.b64encode(salt).decode(),
            'nonce': base64.b64encode(nonce).decode(),
            'tag': base64.b64encode(encryptor.tag).decode(),
            'ciphertext': base64.b64encode(ciphertext).decode()
        }

    def decrypt_vault(self, encrypted, master_password):
        """Decrypt the vault. Raises ValueError on wrong password."""
        salt = base64.b64decode(encrypted['salt'])
        nonce = base64.b64decode(encrypted['nonce'])
        tag = base64.b64decode(encrypted['tag'])
        ciphertext = base64.b64decode(encrypted['ciphertext'])

        key, _ = self.derive_key(master_password, salt)

        cipher = Cipher(algorithms.AES(key), modes.GCM(nonce, tag),
                        backend=self.backend)
        decryptor = cipher.decryptor()
        try:
            plaintext = decryptor.update(ciphertext) + decryptor.finalize()
            return plaintext.decode()
        except InvalidTag:
            raise ValueError("Wrong master password")

Step 2: Password Generator with Entropy Estimation

Generate cryptographically secure random passwords using secrets (not random). Enforce minimum complexity by guaranteeing at least one char from each selected set.

import string
import math

class PasswordGenerator:
    LOWERCASE = string.ascii_lowercase
    UPPERCASE = string.ascii_uppercase
    DIGITS = string.digits
    SYMBOLS = "!@#$%^&*()-_=+[]{}|;:,.<>?/"

    @staticmethod
    def generate(length=20, upper=True, digits=True, symbols=True):
        """Generate a secure random password."""
        pool = PasswordGenerator.LOWERCASE
        required = []

        if upper:
            pool += PasswordGenerator.UPPERCASE
            required.append(secrets.choice(PasswordGenerator.UPPERCASE))
        if digits:
            pool += PasswordGenerator.DIGITS
            required.append(secrets.choice(PasswordGenerator.DIGITS))
        if symbols:
            pool += PasswordGenerator.SYMBOLS
            required.append(secrets.choice(PasswordGenerator.SYMBOLS))

        remaining = [secrets.choice(pool) for _ in range(length - len(required))]
        password = list(required + remaining)
        secrets.SystemRandom().shuffle(password)
        return ''.join(password)

    @staticmethod
    def entropy(password):
        """Calculate bits of entropy."""
        chars = 0
        if any(c.islower() for c in password): chars += 26
        if any(c.isupper() for c in password): chars += 26
        if any(c.isdigit() for c in password): chars += 10
        if any(c in PasswordGenerator.SYMBOLS for c in password):
            chars += len(PasswordGenerator.SYMBOLS)
        return round(len(password) * math.log2(max(chars, 1)), 2)

# Example
pw = PasswordGenerator.generate(length=24)
print(f"Generated: {pw}")
print(f"Entropy: {PasswordGenerator.entropy(pw)} bits")

Step 3: SQLite Vault Storage

Store credentials in SQLite. The entire database file is encrypted at rest. When unlocked, credentials live in memory as Python objects.

import sqlite3
from datetime import datetime

class VaultDB:
    def __init__(self, db_path='vault.db'):
        self.conn = sqlite3.connect(db_path)
        self.conn.row_factory = sqlite3.Row
        self._init_schema()

    def _init_schema(self):
        self.conn.execute('''
            CREATE TABLE IF NOT EXISTS credentials (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                service TEXT NOT NULL,
                username TEXT NOT NULL,
                password TEXT NOT NULL,
                url TEXT,
                notes TEXT,
                created_at TEXT DEFAULT (datetime('now')),
                updated_at TEXT DEFAULT (datetime('now'))
            )
        ''')
        self.conn.execute('''
            CREATE TABLE IF NOT EXISTS vault_meta (
                key TEXT PRIMARY KEY,
                value TEXT NOT NULL
            )
        ''')
        self.conn.commit()

    def add_credential(self, service, username, password, url=None, notes=None):
        self.conn.execute(
            'INSERT INTO credentials (service, username, password, url, notes) VALUES (?,?,?,?,?)',
            (service, username, password, url, notes)
        )
        self.conn.commit()
        return self.conn.execute('SELECT last_insert_rowid()').fetchone()[0]

    def get_credential(self, credential_id):
        return self.conn.execute(
            'SELECT * FROM credentials WHERE id = ?', (credential_id,)
        ).fetchone()

    def search(self, query):
        q = f'%{query}%'
        return self.conn.execute(
            'SELECT * FROM credentials WHERE service LIKE ? OR username LIKE ? OR url LIKE ?',
            (q, q, q)
        ).fetchall()

    def list_all(self):
        return self.conn.execute(
            'SELECT id, service, username, url FROM credentials ORDER BY service'
        ).fetchall()

    def delete(self, credential_id):
        self.conn.execute('DELETE FROM credentials WHERE id = ?', (credential_id,))
        self.conn.commit()

Note: In this design, the vault encryption happens at the file level (you encrypt the entire vault.db file or a JSON export). For simplicity in this tutorial, we encrypt a JSON export of the database contents.

Step 4: CLI Interface

Build a clean, interactive CLI that prompts for the master password once per session.

import argparse
import getpass
import json
import sys
import os

VAULT_FILE = 'vault.enc'
META_FILE = 'vault_meta.json'

def main():
    parser = argparse.ArgumentParser(description='Secure Password Manager')
    sub = parser.add_subparsers(dest='cmd', required=True)

    sub.add_parser('init', help='Initialize a new vault')
    sub.add_parser('unlock', help='Unlock and enter interactive mode')
    sub.add_parser('add', help='Add a credential')
    sub.add_parser('list', help='List all credentials')
    sub.add_parser('search', help='Search credentials')
    sub.add_parser('generate', help='Generate a strong password')

    args = parser.parse_args()
    crypto = CryptoEngine()

    if args.cmd == 'init':
        if os.path.exists(VAULT_FILE):
            print("Vault already exists!")
            return
        master = getpass.getpass("Set master password: ")
        confirm = getpass.getpass("Confirm: ")
        if master != confirm:
            print("Passwords don't match!")
            return
        # Create empty vault and encrypt
        h = crypto.hash_master_password(master)
        empty = json.dumps({'credentials': []})
        enc = crypto.encrypt_vault(empty, master)
        with open(VAULT_FILE, 'w') as f:
            json.dump(enc, f)
        with open(META_FILE, 'w') as f:
            json.dump({'master_hash': h}, f)
        print("✅ Vault initialized!")

    elif args.cmd == 'generate':
        pw = PasswordGenerator.generate()
        print(f"Generated password: {pw}")
        print(f"Entropy: {PasswordGenerator.entropy(pw)} bits")

    elif args.cmd in ('list', 'add', 'search'):
        # Need to unlock first
        if not os.path.exists(VAULT_FILE):
            print("No vault found. Run 'init' first.")
            return
        master = getpass.getpass("Master password: ")
        with open(VAULT_FILE) as f:
            enc = json.load(f)
        try:
            data = json.loads(crypto.decrypt_vault(enc, master))
        except ValueError:
            print("❌ Wrong master password!")
            return

        if args.cmd == 'list':
            for cred in data['credentials']:
                print(f"  [{cred['id']}] {cred['service']} — {cred['username']}")

        elif args.cmd == 'add':
            service = input("Service: ")
            username = input("Username: ")
            password = getpass.getpass("Password (blank to generate): ")
            if not password:
                password = PasswordGenerator.generate()
                print(f"Generated: {password}")
            data['credentials'].append({
                'id': len(data['credentials']) + 1,
                'service': service,
                'username': username,
                'password': password
            })
            # Re-encrypt and save
            new_enc = crypto.encrypt_vault(json.dumps(data), master)
            with open(VAULT_FILE, 'w') as f:
                json.dump(new_enc, f)
            print(f"✅ Credential saved for {service}")

if __name__ == '__main__':
    main()
