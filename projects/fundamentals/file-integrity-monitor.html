Step 1: File Hashing Engine

Build the core hashing function. SHA-256 is the standard — avoid MD5 and SHA-1 for security purposes as they have known collision vulnerabilities.

import hashlib
import os
from pathlib import Path

def hash_file(filepath, algorithm='sha256'):
    """Return hex digest of a file using the specified algorithm."""
    hasher = hashlib.new(algorithm)

    try:
        with open(filepath, 'rb') as f:
            # Read in 64KB chunks to handle files of any size
            while chunk := f.read(65536):
                hasher.update(chunk)
        return hasher.hexdigest()
    except (PermissionError, FileNotFoundError) as e:
        print(f"  [!] Skipping {filepath}: {e}")
        return None

def hash_directory(directory, algorithm='sha256'):
    """Walk a directory and hash every file. Returns {rel_path: hash}."""
    file_hashes = {}
    dir_path = Path(directory)

    for filepath in dir_path.rglob('*'):
        if filepath.is_file():
            file_hash = hash_file(str(filepath), algorithm)
            if file_hash:
                rel = str(filepath.relative_to(dir_path))
                file_hashes[rel] = file_hash

    return file_hashes

Why chunked reading? Reading large files in 64KB blocks prevents memory exhaustion. A 4GB file gets hashed without ever loading more than 64KB into RAM.

Step 2: Baseline Storage & Comparison

Store baseline hashes in JSON and implement the diff logic to detect three types of changes.

import json
from datetime import datetime

BASELINE_FILE = "baseline.json"

def create_baseline(directory, label=None):
    """Hash a directory and save the snapshot."""
    if label is None:
        label = f"baseline-{datetime.now().strftime('%Y%m%d-%H%M%S')}"

    hashes = hash_directory(directory)

    snapshot = {
        "label": label,
        "directory": str(Path(directory).resolve()),
        "timestamp": datetime.now().isoformat(),
        "algorithm": "sha256",
        "file_count": len(hashes),
        "hashes": hashes
    }

    # Load existing baselines, append, and save
    baselines = load_baselines()
    baselines[label] = snapshot
    save_baselines(baselines)

    print(f"Baseline '{label}' created — {len(hashes)} files indexed")
    return snapshot

def load_baselines():
    """Load all baselines from disk."""
    if os.path.exists(BASELINE_FILE):
        with open(BASELINE_FILE) as f:
            return json.load(f)
    return {}

def save_baselines(data):
    with open(BASELINE_FILE, 'w') as f:
        json.dump(data, f, indent=2)

Step 3: Change Detection Engine

Compare the current file state against a stored baseline to find modified, added, and deleted files.

def scan_changes(directory, baseline_label):
    """Compare current directory state to a baseline snapshot."""
    baselines = load_baselines()

    if baseline_label not in baselines:
        print(f"No baseline named '{baseline_label}'")
        print(f"Available: {', '.join(baselines.keys())}")
        return

    baseline = baselines[baseline_label]
    baseline_hashes = baseline["hashes"]
    current_hashes = hash_directory(directory)

    # Detect changes
    modified = []  # file exists in both, hash differs
    added = []     # file exists now, not in baseline
    deleted = []   # file was in baseline, not on disk now

    for path, old_hash in baseline_hashes.items():
        if path in current_hashes:
            if current_hashes[path] != old_hash:
                modified.append(path)
        else:
            deleted.append(path)

    for path in current_hashes:
        if path not in baseline_hashes:
            added.append(path)

    # Generate report
    total = len(modified) + len(added) + len(deleted)
    print(f"\n{'='*55}")
    print(f"  FILE INTEGRITY SCAN — {baseline_label}")
    print(f"  Baseline: {baseline['timestamp']}")
    print(f"{'='*55}")

    if total == 0:
        print("  ✓ No changes detected — system integrity verified")
        return

    print(f"  ⚠ {total} change(s) detected:\n")

    if modified:
        print(f"  [MODIFIED] {len(modified)} file(s):")
        for f in modified:
            print(f"    • {f}")
    if added:
        print(f"  [ADDED]    {len(added)} file(s):")
        for f in added:
            print(f"    • {f}")
    if deleted:
        print(f"  [DELETED]  {len(deleted)} file(s):")
        for f in deleted:
            print(f"    • {f}")

    print(f"{'='*55}\n")

Detection logic: Three categories cover every possible state transition. Modified = same path, different hash. Added = new path. Deleted = path missing.

Step 4: Complete CLI Tool

Wire everything together with argparse subcommands for a professional interface.

import argparse

def main():
    parser = argparse.ArgumentParser(
        description="File Integrity Monitor — detect unauthorized changes"
    )
    sub = parser.add_subparsers(dest="cmd", help="Commands")

    # baseline command
    bl = sub.add_parser("baseline", help="Create a new baseline")
    bl.add_argument("directory", help="Directory to monitor")
    bl.add_argument("-n", "--name", help="Baseline label (auto-generated if omitted)")

    # scan command
    sc = sub.add_parser("scan", help="Scan for changes against a baseline")
    sc.add_argument("directory", help="Directory to scan")
    sc.add_argument("-b", "--baseline", required=True, help="Baseline label to compare against")

    # list command
    ls = sub.add_parser("list", help="List all baselines")

    args = parser.parse_args()

    if args.cmd == "baseline":
        create_baseline(args.directory, args.name)
    elif args.cmd == "scan":
        scan_changes(args.directory, args.baseline)
    elif args.cmd == "list":
        baselines = load_baselines()
        if not baselines:
            print("No baselines found. Create one with: fim baseline ")
        else:
            for label, data in baselines.items():
                print(f"  {label} — {data['file_count']} files ({data['timestamp']})")
    else:
        parser.print_help()

if __name__ == "__main__":
    main()
