Step 1: Caesar Cipher Encryption & Decryption

The Caesar cipher shifts each letter by a fixed number of positions. With shift=3, 'A' becomes 'D', 'B' becomes 'E', and so on. It wraps around at 'Z' back to 'A'. This is modular arithmetic: new_char = (char_pos + shift) % 26.

def caesar_encrypt(plaintext, shift):
    """Encrypt using Caesar cipher — shifts letters forward by `shift`."""
    result = []
    for char in plaintext:
        if char.isupper():
            result.append(chr((ord(char) - ord('A') + shift) % 26 + ord('A')))
        elif char.islower():
            result.append(chr((ord(char) - ord('a') + shift) % 26 + ord('a')))
        else:
            result.append(char)  # Preserve non-letters
    return ''.join(result)

def caesar_decrypt(ciphertext, shift):
    """Decrypt by shifting backward — equivalent to encrypting with (26 - shift)."""
    return caesar_encrypt(ciphertext, 26 - shift)  # Or pass -shift

# Test with ROT13 (shift=13, self-inverse)
original = "HELLO WORLD"
encrypted = caesar_encrypt(original, 13)
decrypted = caesar_decrypt(encrypted, 13)

print(f"Original:  {original}")
print(f"Encrypted: {encrypted}")
print(f"Decrypted: {decrypted}")

# Output:
# Original:  HELLO WORLD
# Encrypted: URYYB JBEYQ
# Decrypted: HELLO WORLD

Key insight: The Caesar cipher has only 25 possible keys (shifts 1–25, since 0 and 26 are identity). That's trivially brute-forceable — a lesson in why key space size matters.

Step 2: Brute-Force Cracker

Since there are only 25 possible shifts, we can try them all and print every result. A human (or an automated scorer) can pick the correct one.

def brute_force_caesar(ciphertext):
    """Try all 25 shifts and display results."""
    print(f"{'Shift':>5} | Decryption")
    print("-" * 40)
    for shift in range(1, 26):
        plaintext = caesar_decrypt(ciphertext, shift)
        print(f"{shift:>5} | {plaintext}")

# Try it on an intercepted message
intercepted = "LIW QI XLMHI WIGVIX QC TIRWIV"
brute_force_caesar(intercepted)

# Output:
# Shift | Decryption
# ----------------------------------------
#     1 | KHV PH WKLGH VHFUHW PB UHQHU
#     ...
#     4 | HELLO WORLD — THIS IS THE CORRECT ONE!
#     ...
#    13 | YVD FV KYSVY DVTVIE HZ EFIZFI

Step 3: Frequency Analysis — Cracking Without Trying All Shifts

In English, 'E' is the most common letter (~12.7%), followed by 'T', 'A', 'O', etc. If we find the most common letter in ciphertext, we can calculate the likely shift.

from collections import Counter
import string

# Standard English letter frequencies (simplified)
ENGLISH_FREQ = {
    'E': 12.70, 'T': 9.06, 'A': 8.17, 'O': 7.51, 'I': 6.97,
    'N': 6.75, 'S': 6.33, 'H': 6.09, 'R': 5.99, 'D': 4.25,
    'L': 4.03, 'C': 2.78, 'U': 2.76, 'M': 2.41, 'W': 2.36,
    'F': 2.23, 'G': 2.02, 'Y': 1.97, 'P': 1.93, 'B': 1.49,
    'V': 0.98, 'K': 0.77, 'J': 0.15, 'X': 0.15, 'Q': 0.10, 'Z': 0.07
}

def frequency_analysis_crack(ciphertext):
    """Crack Caesar cipher by comparing frequency distributions."""
    # Count letters in ciphertext
    letters_only = [c.upper() for c in ciphertext if c.isalpha()]
    if not letters_only:
        return None

    counter = Counter(letters_only)
    total = sum(counter.values())

    # Compare each possible shift against English frequencies
    best_shift, best_score = 0, float('inf')

    for shift in range(26):
        score = 0
        for plain_char in string.ascii_uppercase:
            # What ciphertext character maps to this plain char under this shift?
            cipher_char = chr((ord(plain_char) - ord('A') + shift) % 26 + ord('A'))
            observed_pct = (counter.get(cipher_char, 0) / total) * 100
            expected_pct = ENGLISH_FREQ.get(plain_char, 0)
            score += abs(observed_pct - expected_pct)

        if score < best_score:
            best_score = score
            best_shift = shift

    return caesar_decrypt(ciphertext, best_shift), best_shift

# Crack it
ct = "LIW QI XLMHI WIGVIX QC TIRWIV"
plaintext, shift = frequency_analysis_crack(ct)
print(f"Likely shift: {shift}")
print(f"Cracked text: {plaintext}")
# Output:
# Likely shift: 4
# Cracked text: HELLO WORLD — THIS IS THE CORRECT ONE!

Frequency analysis doesn't just break Caesar ciphers — it breaks all monoalphabetic substitution ciphers. This is why modern ciphers (AES) produce output that looks uniformly random, with no detectable frequency patterns.

Step 4: Beyond Caesar — ROT13 and Custom Alphabets

Extend your tool to support ROT13 (used historically for obscuring text on forums) and custom substitution alphabets.

class SubstitutionCipher:
    """General substitution cipher with custom alphabet mapping."""

        if alphabet is None:

            alpha_list = list(string.ascii_uppercase)
            secrets.SystemRandom().shuffle(alpha_list)
            alphabet = ''.join(alpha_list)

        self.encode_map = {}
        self.decode_map = {}
        for plain, cipher in zip(string.ascii_uppercase, alphabet):
            self.encode_map[plain] = cipher
            self.decode_map[cipher] = plain

    def encrypt(self, text):
        result = []
        for c in text:
            upper = c.upper()
            if upper in self.encode_map:
                mapped = self.encode_map[upper]
                result.append(mapped if c.isupper() else mapped.lower())
            else:
                result.append(c)
        return ''.join(result)

    def decrypt(self, text):
        result = []
        for c in text:
            upper = c.upper()
            if upper in self.decode_map:
                mapped = self.decode_map[upper]
                result.append(mapped if c.isupper() else mapped.lower())
            else:
                result.append(c)
        return ''.join(result)
