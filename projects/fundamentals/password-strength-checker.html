Password Strength Checker

Build a tool that evaluates password strength using entropy calculation, pattern detection, and common password checks.

Learning Objectives

  • Understand password entropy and how it quantifies password strength mathematically
  • Learn to use Python's re module for pattern matching and validation
  • Detect common weak password patterns (dictionary words, sequences, keyboard walks)
  • Calculate character set diversity and its impact on brute-force resistance
  • Provide actionable feedback that helps users create stronger passwords

Prerequisites

  • Python 3.8+ installed
  • Basic understanding of Python strings, lists, and functions
  • Familiarity with regular expressions (basic level)

Step 1: Core Validation Functions

Start by building the core password validation functions. These check length, character diversity, and common patterns.

import re
import math

def check_length(password, min_length=8):
    """Check if password meets minimum length."""
    if len(password) >= min_length:
        return True, f"Length: {len(password)} characters (OK)"
    return False, f"Too short — need at least {min_length} characters"

def check_character_types(password):
    """Count character set categories present."""
    has_lower = bool(re.search(r'[a-z]', password))
    has_upper = bool(re.search(r'[A-Z]', password))
    has_digit = bool(re.search(r'\d', password))
    has_special = bool(re.search(r'[^a-zA-Z0-9]', password))

    categories = sum([has_lower, has_upper, has_digit, has_special])
    types_found = []
    if has_lower: types_found.append("lowercase")
    if has_upper: types_found.append("uppercase")
    if has_digit: types_found.append("digits")
    if has_special: types_found.append("special chars")

    return categories, types_found

def check_common_patterns(password):
    """Detect keyboard walks, sequences, and common passwords."""
    common = ['password', '123456', 'qwerty', 'abc123',
              'admin', 'letmein', 'monkey', 'dragon',
              'iloveyou', 'welcome', 'football']
    lower = password.lower()

    for pattern in common:
        if pattern in lower:
            return False, f"Contains common word: '{pattern}'"

    # Check for sequential characters
    for i in range(len(password) - 2):
        a, b, c = ord(password[i]), ord(password[i+1]), ord(password[i+2])
        if (b - a == 1 and c - b == 1) or (a - b == 1 and b - c == 1):
            return False, "Contains sequential characters"

    # Check for repeated characters (3+)
    if re.search(r'(.)\1{2,}', password):
        return False, "Contains repeated characters (3+)"

    return True, "No common patterns detected"

Key concept: Each function isolates one aspect of password strength. This modular approach makes it easy to add new checks later without touching existing logic.

Step 2: Entropy Calculation

Entropy measures true password strength in bits. Higher entropy means exponentially harder to crack.

def calculate_entropy(password):
    """Calculate Shannon entropy and estimated crack time."""
    # Determine pool size (characters available)
    pool = 0
    if re.search(r'[a-z]', password):
        pool += 26
    if re.search(r'[A-Z]', password):
        pool += 26
    if re.search(r'\d', password):
        pool += 10
    if re.search(r'[^a-zA-Z0-9]', password):
        pool += 32  # common special chars

    if pool == 0:
        return 0, "N/A"

    # Entropy = log2(pool ** length) = length * log2(pool)
    entropy = len(password) * math.log2(pool)

    # Estimate crack time (assume 10^12 guesses/sec — high-end GPU farm)
    guesses = 2 ** entropy
    crack_seconds = guesses / 1e12

    time_str = format_time(crack_seconds)
    return round(entropy, 1), time_str

def format_time(seconds):
    """Convert seconds to human-readable time."""
    if seconds < 1:
        return "instantly"
    elif seconds < 60:
        return f"{seconds:.0f} seconds"
    elif seconds < 3600:
        return f"{seconds/60:.1f} minutes"
    elif seconds < 86400:
        return f"{seconds/3600:.1f} hours"
    elif seconds < 31536000:
        return f"{seconds/86400:.0f} days"
    elif seconds < 31536000 * 100:
        return f"{seconds/31536000:.0f} years"
    else:
        return "centuries"

Math behind it: A password using 4 character sets has a pool of 94 characters. At 12 characters, that's 9412 = ~279 combinations (79 bits of entropy).

Step 3: Scoring Engine & Reporting

Combine all checks into a weighted scoring system and produce a clear report.

def analyze_password(password):
    """Full password analysis — returns score and detailed report."""
    score = 0
    report = []

    # Length (30 pts max)
    length_ok, length_msg = check_length(password)
    if length_ok:
        length_bonus = min(30, (len(password) - 8) * 5 + 10)
        score += length_bonus
    report.append(("Length", length_msg, length_ok))

    # Character diversity (30 pts max)
    cats, types = check_character_types(password)
    score += cats * 8  # 8 pts per category (max 32, capped at 30)
    report.append(("Character Types", f"Found: {', '.join(types)}", cats >= 3))

    # Pattern check (20 pts)
    pattern_ok, pattern_msg = check_common_patterns(password)
    if pattern_ok:
        score += 20
    report.append(("Patterns", pattern_msg, pattern_ok))

    # Entropy (20 pts max)
    entropy, crack_time = calculate_entropy(password)
    entropy_score = min(20, int(entropy / 4))
    score += entropy_score
    report.append(("Entropy", f"{entropy} bits — cracked in {crack_time}", entropy >= 60))

    # Clamp final score
    score = min(100, score)

    # Rating
    if score >= 80:
        rating = "Strong"
    elif score >= 60:
        rating = "Moderate"
    elif score >= 40:
        rating = "Weak"
    else:
        rating = "Very Weak"

    return {
        "score": score,
        "rating": rating,
        "entropy": entropy,
        "crack_time": crack_time,
        "report": report
    }

Step 4: Command-Line Interface

Build a clean CLI that accepts passwords and displays detailed results.

import argparse
import sys

def main():
    parser = argparse.ArgumentParser(
        description="Password Strength Checker — evaluate password security"
    )
    parser.add_argument("password", nargs="?", help="Password to analyze")
    parser.add_argument("-f", "--file", help="Check passwords from a file (one per line)")
    parser.add_argument("-q", "--quiet", action="store_true",
                        help="Output only score (for scripting)")
    args = parser.parse_args()

    passwords = []
    if args.file:
        with open(args.file) as f:
            passwords = [line.strip() for line in f if line.strip()]
    elif args.password:
        passwords = [args.password]
    else:
        import getpass
        passwords = [getpass.getpass("Enter password to check: ")]

    for pwd in passwords:
        result = analyze_password(pwd)
        if args.quiet:
            print(result["score"])
            continue

        bar = "█" * (result["score"] // 5) + "░" * (20 - result["score"] // 5)
        print(f"\n{'='*50}")
        print(f"  PASSWORD STRENGTH: {result['rating']}")
        print(f"  [{bar}] {result['score']}/100")
        print(f"  Entropy: {result['entropy']} bits")
        print(f"  Estimated crack time: {result['crack_time']}")
        print(f"\n  Details:")
        for name, msg, ok in result["report"]:
            icon = "✓" if ok else "✗"
            print(f"  {icon} {name}: {msg}")

if __name__ == "__main__":
    main()
